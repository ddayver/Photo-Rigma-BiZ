890 private function calculate_thumbnail_size(array $size): array
891 {
892 // Валидация параметра $size
893 if (!isset($size[0], $size[1]) ||
894 !is_int($size[0]) || !is_int($size[1]) ||
895 $size[0] <= 0 || $size[1] <= 0) {
896 throw new \InvalidArgumentException(
897 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректные размеры изображения | " .
898 "Требуется массив с двумя положительными целыми числами. Получено: width = " . (isset($size[0]) ? $size[0] : 'undefined') . ", height = " . (isset($size[1]) ? $size[1] : 'undefined')
899 );
900 }
901
902 // Валидация конфигурации
903 if (!isset($this->config['temp_photo_w'], $this->config['temp_photo_h']) ||
904 !is_numeric($this->config['temp_photo_w']) || !is_numeric($this->config['temp_photo_h']) ||
905 $this->config['temp_photo_w'] < 0 || $this->config['temp_photo_h'] < 0) {
906 throw new \RuntimeException(
907 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректная конфигурация temp_photo_w или temp_photo_h | " .
908 "Значения должны быть числами ≥ 0. Получено: temp_photo_w = " . (isset($this->config['temp_photo_w']) ? $this->config['temp_photo_w'] : 'undefined') . ", temp_photo_h = " . (isset($this->config['temp_photo_h']) ? $this->config['temp_photo_h'] : 'undefined')
909 );
910 }
911
912 // Преобразуем конфигурацию в целые числа
913 $target_width = (int)$this->config['temp_photo_w'];
914 $target_height = (int)$this->config['temp_photo_h'];
915
916 // Вычисляем коэффициенты масштабирования через match
917 $ratio_width = match (true) {
918 $target_width > 0 => $size[0] / $target_width,
919 default => 1,
920 };
921
922 $ratio_height = match (true) {
923 $target_height > 0 => $size[1] / $target_height,
924 default => 1,
925 };
926
927 // Если исходное изображение меньше целевого размера, возвращаем оригинальные размеры
928 if ($size[0] < $target_width && $size[1] < $target_height && $target_width > 0 && $target_height > 0) {
929 return [
930 'width' => $size[0],
931 'height' => $size[1]
932 ];
933 }
934
935 // Масштабируем изображение пропорционально
936 if ($ratio_width < $ratio_height) {
937 return [
938 'width' => (int)($size[0] / $ratio_height),
939 'height' => (int)($size[1] / $ratio_height)
940 ];
941 } else {
942 return [
943 'width' => (int)($size[0] / $ratio_width),
944 'height' => (int)($size[1] / $ratio_width)
945 ];
946 }
947 }

1014 private function process_image_resize_gmagick(array $original_data, array $thumbnail_data): bool
1015 {
1016 // Разрешённые MIME-типы и их соответствие форматам GraphicsMagick
1017 $gmagick_formats = [
1018 'image/jpeg' => 'JPEG',
1019 'image/png' => 'PNG',
1020 'image/gif' => 'GIF',
1021 'image/webp' => 'WEBP',
1022 'image/tiff' => 'TIFF',
1023 'image/svg+xml' => 'SVG',
1024 'image/bmp' => 'BMP',
1025 'image/x-icon' => 'ICO', // Иконки
1026 'image/avif' => 'AVIF', // Современный формат
1027 'image/heic' => 'HEIC', // Формат с камер iPhone
1028 ];
1029 // Проверка MIME-типа
1030 if (!array_key_exists($original_data['type'], $gmagick_formats)) {
1031 log_in_file(
1032 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неподдерживаемый MIME-тип для GraphicsMagick | Получено: {$original_data['type']}"
1033 );
1034 return false;
1035 }
1036 // Получение формата GraphicsMagick
1037 $format = $gmagick_formats[$original_data['type']];
1038 // Проверка поддержки формата через GraphicsMagick
1039 $supported_formats = \Gmagick::queryFormats();
1040 if (!in_array($format, $supported_formats)) {
1041 log_in_file(
1042 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Формат {$format} не поддерживается GraphicsMagick"
1043 );
1044 return false;
1045 }
1046 // Ограничение памяти
1047 $available_memory = (int)ini_get('memory_limit'); // Например, '128M' → 128 * 1024 * 1024
1048 $max_memory_usage = 0.25 * $available_memory; // 25% от доступной памяти
1049 $estimated_memory = $original_data['width'] * $original_data['height'] * 3;
1050 if ($estimated_memory > $max_memory_usage) {
1051 log_in_file(
1052 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Изображение слишком велико для обработки через GraphicsMagick"
1053 );
1054 return false;
1055 }
1056 // Создание резервной копии старого эскиза, если он существует
1057 if ($thumbnail_data['exists']) {
1058 $backup_thumbnail_path = tempnam(dirname($thumbnail_data['path']), 'bak_');
1059 if (!$backup_thumbnail_path || !rename($thumbnail_data['path'], $backup_thumbnail_path)) {
1060 log_in_file(
1061 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось создать резервную копию старого эскиза через GraphicsMagick"
1062 );
1063 return false;
1064 }
1065 } else {
1066 $backup_thumbnail_path = null;
1067 }
1068 $image = null;
1069 try {
1070 // Создание объекта Gmagick
