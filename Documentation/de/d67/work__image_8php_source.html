813 ])) {

815 }

818 throw new \InvalidArgumentException(
819 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный формат пути | Ошибки: " . implode(' ', $path_errors)

823 if (!file_exists($full_path) || !is_readable($full_path)) {
824 throw new \RuntimeException(
825 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Исходное изображение не найдено или недоступно для чтения | Путь: \$full_path = {$full_path}"
826 );
827 }
828 // Проверка доступности директории для записи
829 $thumbnail_dir = dirname($thumbnail_path);
830 if (!is_writable($thumbnail_dir)) {
831 throw new \RuntimeException(
832 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Директория для сохранения эскиза недоступна для записи | Путь: \$thumbnail_dir = {$thumbnail_dir}"
833 );
834 }

836 if ($full_size === false) {
837 throw new \RuntimeException(

847 if ($thumbnail_exists && !is_writable($thumbnail_path)) {

850 );
851 }
852 // Блок 3: Расчет размеров будущего эскиза
853 $photo = $this->calculate_thumbnail_size($full_size);
854 // Если файл эскиза существует и его размеры совпадают с расчетными, завершаем работу
855 if ($thumbnail_exists) {
856 $thumbnail_size = getimagesize($thumbnail_path);

862 $finfo = finfo_open(FILEINFO_MIME_TYPE);
863 $mime_type = finfo_file($finfo, $full_path);
864 finfo_close($finfo);
865 // Блок 5: Подготовка данных для обработки изображения
866 $original_data = [
867 'path' => $full_path,
868 'type' => $mime_type,
869 'width' => $full_size[0],
870 'height' => $full_size[1],
871 ];
872 $thumbnail_data = [
873 'path' => $thumbnail_path,
874 'width' => $photo['width'],

876 'exists' => $thumbnail_exists,
877 ];

886 );

895 );
896 }
897 // Если ни одна из библиотек не сработала, используем GD

1224 private function calculate_thumbnail_size(array $size): array
1225 {
1226 // Валидация параметра $size
1227 if (!isset($size[0], $size[1]) ||
1228 !is_int($size[0]) || !is_int($size[1]) ||
1229 $size[0] <= 0 || $size[1] <= 0) {
1230 throw new \InvalidArgumentException(
1231 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректные размеры изображения | " .
1232 "Требуется массив с двумя положительными целыми числами. Получено: width = " . (isset($size[0]) ? $size[0] : 'undefined') . ", height = " . (isset($size[1]) ? $size[1] : 'undefined')
1233 );
1234 }
1235
1236 // Валидация конфигурации
1237 if (!isset($this->config['temp_photo_w'], $this->config['temp_photo_h']) ||
1238 !is_numeric($this->config['temp_photo_w']) || !is_numeric($this->config['temp_photo_h']) ||
1239 $this->config['temp_photo_w'] < 0 || $this->config['temp_photo_h'] < 0) {
1240 throw new \RuntimeException(
1241 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректная конфигурация temp_photo_w или temp_photo_h | " .
1242 "Значения должны быть числами ≥ 0. Получено: temp_photo_w = " . (isset($this->config['temp_photo_w']) ? $this->config['temp_photo_w'] : 'undefined') . ", temp_photo_h = " . (isset($this->config['temp_photo_h']) ? $this->config['temp_photo_h'] : 'undefined')
1243 );
1244 }
1245
1246 // Преобразуем конфигурацию в целые числа
1247 $target_width = (int)$this->config['temp_photo_w'];
1248 $target_height = (int)$this->config['temp_photo_h'];
1249
1250 // Вычисляем коэффициенты масштабирования через match
1251 $ratio_width = match (true) {
1252 $target_width > 0 => $size[0] / $target_width,
1253 default => 1,
1254 };
1255
1256 $ratio_height = match (true) {
1257 $target_height > 0 => $size[1] / $target_height,
1258 default => 1,
1259 };
1260
1261 // Если исходное изображение меньше целевого размера, возвращаем оригинальные размеры
1262 if ($size[0] < $target_width && $size[1] < $target_height && $target_width > 0 && $target_height > 0) {
1263 return [
1264 'width' => $size[0],
1265 'height' => $size[1]
1266 ];
1267 }
1268
1269 // Масштабируем изображение пропорционально
1270 if ($ratio_width < $ratio_height) {
1271 return [
1272 'width' => (int)($size[0] / $ratio_height),
1273 'height' => (int)($size[1] / $ratio_height)
1274 ];
1275 } else {
1276 return [
1277 'width' => (int)($size[0] / $ratio_width),
1278 'height' => (int)($size[1] / $ratio_width)
1279 ];
1280 }
1281 }

1348 private function process_image_resize_gmagick(array $original_data, array $thumbnail_data): bool

1353 'image/png' => 'PNG',

1355 'image/webp' => 'WEBP',
1356 'image/tiff' => 'TIFF',

1363 // Проверка MIME-типа
1364 if (!array_key_exists($original_data['type'], $gmagick_formats)) {
1365 log_in_file(

1372 // Проверка поддержки формата через GraphicsMagick
1373 $supported_formats = \Gmagick::queryFormats();

1376 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Формат {$format} не поддерживается GraphicsMagick"
1377 );

1382 $max_memory_usage = 0.25 * $available_memory; // 25% от доступной памяти
1383 $estimated_memory = $original_data['width'] * $original_data['height'] * 3;
1384 if ($estimated_memory > $max_memory_usage) {
1385 log_in_file(
1386 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Изображение слишком велико для обработки через GraphicsMagick"
1387 );

1389 }
1390 // Создание резервной копии старого эскиза, если он существует
1391 if ($thumbnail_data['exists']) {
1392 $backup_thumbnail_path = tempnam(dirname($thumbnail_data['path']), 'bak_');
