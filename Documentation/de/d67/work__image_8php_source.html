710 private function calculate_thumbnail_size(array $size): array
711 {
712 // Валидация параметра $size
713 if (!isset($size[0], $size[1]) || !is_int($size[0]) || !is_int($size[1]) || $size[0] <= 0 || $size[1] <= 0) {
714 throw new InvalidArgumentException(
715 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректные размеры изображения | " . "Требуется массив с двумя положительными целыми числами. Получено: width = " . (isset($size[0]) ? $size[0] : 'undefined') . ", height = " . (isset($size[1]) ? $size[1] : 'undefined')
716 );
717 }
718
719 // Валидация конфигурации
720 if (!isset($this->config['temp_photo_w'], $this->config['temp_photo_h']) || !is_numeric(
721 $this->config['temp_photo_w']
722 ) || !is_numeric(
723 $this->config['temp_photo_h']
724 ) || $this->config['temp_photo_w'] < 0 || $this->config['temp_photo_h'] < 0) {
725 throw new RuntimeException(
726 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректная конфигурация temp_photo_w или temp_photo_h | " . "Значения должны быть числами ≥ 0. Получено: temp_photo_w = " . (isset($this->config['temp_photo_w']) ? $this->config['temp_photo_w'] : 'undefined') . ", temp_photo_h = " . (isset($this->config['temp_photo_h']) ? $this->config['temp_photo_h'] : 'undefined')
727 );
728 }
729
730 // Преобразуем конфигурацию в целые числа
731 $target_width = (int)$this->config['temp_photo_w'];
732 $target_height = (int)$this->config['temp_photo_h'];
733
734 // Вычисляем коэффициенты масштабирования через match
735 $ratio_width = match (true) {
736 $target_width > 0 => $size[0] / $target_width,
737 default => 1,
738 };
739
740 $ratio_height = match (true) {
741 $target_height > 0 => $size[1] / $target_height,
742 default => 1,
743 };
744
745 // Если исходное изображение меньше целевого размера, возвращаем оригинальные размеры
746 if ($size[0] < $target_width && $size[1] < $target_height && $target_width > 0 && $target_height > 0) {
747 return [
748 'width' => $size[0],
749 'height' => $size[1]
750 ];
751 }
752
753 // Масштабируем изображение пропорционально
754 if ($ratio_width < $ratio_height) {
755 return [
756 'width' => (int)($size[0] / $ratio_height),
757 'height' => (int)($size[1] / $ratio_height)
758 ];
759 } else {
760 return [
761 'width' => (int)($size[0] / $ratio_width),
762 'height' => (int)($size[1] / $ratio_width)
763 ];
764 }

872 protected function _image_resize_internal(string $full_path, string $thumbnail_path): bool
873 {

875 $full_path = realpath($full_path);
876 $thumbnail_path = realpath($thumbnail_path);
877 if (!$full_path || !$thumbnail_path) {
878 throw new InvalidArgumentException(
879 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный путь к файлу | \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path}"
880 );
881 }
882 // Блок 1: Проверка корректности путей через filter_var
883 $path_errors = [];
884 foreach (['full_path' => $full_path, 'thumbnail_path' => $thumbnail_path] as $key => $path) {
885 if (!filter_var($path, FILTER_VALIDATE_REGEXP, [
886 'options' => ['regexp' => '/^[a-zA-Z0-9\/\.\-_]+$/']
887 ])) {
888 $path_errors[] = "Некорректный формат пути: \${$key} = {$path}.";
889 }
890 }
891 if (!empty($path_errors)) {
892 throw new InvalidArgumentException(
893 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный формат пути | Ошибки: " . implode(
894 ' ',
895 $path_errors
896 )
897 );
898 }
899 // Блок 2: Проверка существования и доступности файлов
900 if (!file_exists($full_path) || !is_readable($full_path)) {
901 throw new RuntimeException(
902 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Исходное изображение не найдено или недоступно для чтения | Путь: \$full_path = {$full_path}"
903 );
904 }
905 // Проверка доступности директории для записи
906 $thumbnail_dir = dirname($thumbnail_path);
907 if (!is_writable($thumbnail_dir)) {
908 throw new RuntimeException(
909 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Директория для сохранения эскиза недоступна для записи | Путь: \$thumbnail_dir = {$thumbnail_dir}"
910 );
911 }
912 $full_size = getimagesize($full_path);
913 if ($full_size === false) {
914 throw new RuntimeException(
915 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить размеры изображения | Путь: \$full_path = {$full_path}"
916 );
917 }
918 if ($full_size[0] > self::MAX_IMAGE_WIDTH || $full_size[1] > self::MAX_IMAGE_HEIGHT) {
919 throw new RuntimeException(
920 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размеры исходного изображения слишком велики | Путь: \$full_path = {$full_path}, ширина = {$full_size[0]}, высота = {$full_size[1]}"
921 );
922 }
923 $thumbnail_exists = file_exists($thumbnail_path);
924 if ($thumbnail_exists && !is_writable($thumbnail_path)) {
925 throw new RuntimeException(
926 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Файл эскиза существует, но недоступен для записи | Путь: \$thumbnail_path = {$thumbnail_path}"
927 );
928 }
929 // Блок 3: Расчет размеров будущего эскиза
930 $photo = $this->calculate_thumbnail_size($full_size);
931 // Если файл эскиза существует и его размеры совпадают с расчетными, завершаем работу
932 if ($thumbnail_exists) {
933 $thumbnail_size = getimagesize($thumbnail_path);
934 if ($thumbnail_size !== false && $thumbnail_size[0] == $photo['width'] && $thumbnail_size[1] == $photo['height']) {
935 return true;
936 }
937 }
938 // Блок 4: Определение MIME-типа
939 $finfo = finfo_open(FILEINFO_MIME_TYPE);
940 $mime_type = finfo_file($finfo, $full_path);
941 finfo_close($finfo);
942 // Блок 5: Подготовка данных для обработки изображения
943 $original_data = [
944 'path' => $full_path,
945 'type' => $mime_type,
946 'width' => $full_size[0],
947 'height' => $full_size[1],
948 ];
949 $thumbnail_data = [
950 'path' => $thumbnail_path,
951 'width' => $photo['width'],
952 'height' => $photo['height'],
953 'exists' => $thumbnail_exists,
954 ];
955 // Блок 6: Логика выбора библиотеки с fallback
956 try {
957 if (extension_loaded('gmagick')) {
958 return $this->process_image_resize_gmagick($original_data, $thumbnail_data);
959 }
960 } catch (Exception $e) {
961 log_in_file(
962 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через GraphicsMagick | Аргументы: \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path} | Сообщение об ошибке: {$e->getMessage()}"
963 );
964 }
965 try {
966 if (extension_loaded('imagick')) {
967 return $this->process_image_resize_imagick($original_data, $thumbnail_data);
968 }
969 } catch (Exception $e) {
970 log_in_file(
971 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через ImageMagick | Аргументы: \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path} | Сообщение об ошибке: {$e->getMessage()}"
972 );
973 }
974 // Если ни одна из библиотек не сработала, используем GD
975 return $this->process_image_resize_gd($original_data, $thumbnail_data);
976 }

1043 private function process_image_resize_gmagick(array $original_data, array $thumbnail_data): bool
1044 {
1045 // Разрешённые MIME-типы и их соответствие форматам GraphicsMagick
1046 $gmagick_formats = [

1048 'image/png' => 'PNG',

1056 'image/heic' => 'HEIC', // Формат с камер iPhone
1057 ];

1059 if (!array_key_exists($original_data['type'], $gmagick_formats)) {
1060 log_in_file(
1061 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неподдерживаемый MIME-тип для GraphicsMagick | Получено: {$original_data['type']}"
1062 );
1063 return false;
1064 }
1065 // Получение формата GraphicsMagick
1066 $format = $gmagick_formats[$original_data['type']];
1067 // Проверка поддержки формата через GraphicsMagick
1068 $gmagick_instance = new Gmagick(); // Создаем экземпляр Gmagick
1069 $supported_formats = $gmagick_instance->queryFormats(); // Вызываем метод queryFormats()
1070 unset($gmagick_instance);
1071 if (!in_array($format, $supported_formats)) {
1072 log_in_file(
1073 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Формат {$format} не поддерживается GraphicsMagick"
1074 );
1075 return false;
1076 }
1077 // Ограничение памяти
1078 $available_memory = (int)ini_get('memory_limit') * 1024 * 1024; // Например, '128M' → 128 * 1024 * 1024
1079 $max_memory_usage = 0.25 * $available_memory; // 25% от доступной памяти
1080 $estimated_memory = $original_data['width'] * $original_data['height'] * 3;
1081 if ($estimated_memory > $max_memory_usage) {
1082 log_in_file(
1083 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Изображение слишком велико для обработки через GraphicsMagick | Доступная память: " . ini_get(
1084 'memory_limit'
1085 )
1086 );
1087 return false;
1088 }
1089 // Создание резервной копии старого эскиза, если он существует
1090 if ($thumbnail_data['exists']) {
1091 $backup_thumbnail_path = tempnam(dirname($thumbnail_data['path']), 'bak_');
1092 if (!$backup_thumbnail_path || !rename($thumbnail_data['path'], $backup_thumbnail_path)) {
1093 log_in_file(
1094 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось создать резервную копию старого эскиза через GraphicsMagick"
1095 );
1096 return false;

1098 } else {
1099 $backup_thumbnail_path = null;
1100 }
1101 $image = null;
1102 try {
1103 // Создание объекта Gmagick
1104 $image = new Gmagick($original_data['path']);
1105 // Настройка прозрачности для PNG и WebP
1106 if ($original_data['type'] === 'image/png' || $original_data['type'] === 'image/webp') {
1107 $image->setImageFormat($format);
1108 $image->setImageBackgroundColor('transparent'); // Установка прозрачного фона
1109 // $image = $image->flattenImages(); // Объединение слоев с учетом прозрачности
1110 }

1115 Gmagick::FILTER_LANCZOS, // Высококачественный фильтр
1116 1 // Коэффициент размытия (1 = без размытия)
1117 );
1118 // Сохранение нового эскиза
1119 $image->writeImage($thumbnail_data['path']);
1120 // Удаление резервной копии старого эскиза
1121 if ($backup_thumbnail_path && file_exists($backup_thumbnail_path)) {
1122 unlink($backup_thumbnail_path);
1123 }
1124 return true;
1125 } catch (Exception $e) {
1126 // Логирование ошибки с контекстом
1127 log_in_file(
1128 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при создании эскиза через GraphicsMagick | Оригинальный файл: {$original_data['path']}, Эскиз: {$thumbnail_data['path']}, Сообщение об ошибке: {$e->getMessage()}"
1129 );

1133 }

1135 } finally {
1136 // Освобождение ресурсов
1137 if (isset($image) && $image instanceof Gmagick) {
1138 $image->destroy();
1139 }
