1237 {

1261

1263 $ratio_width = match (true) {
1264 $target_width > 0 => $size[0] / $target_width,
1265 default => 1,
1266 };
1267
1268 $ratio_height = match (true) {
1269 $target_height > 0 => $size[1] / $target_height,
1270 default => 1,
1271 };
1272

1276 'width' => $size[0],
1277 'height' => $size[1]
1278 ];
1279 }
1280
1281 // Масштабируем изображение пропорционально
1282 if ($ratio_width < $ratio_height) {
1283 return [
1284 'width' => (int)($size[0] / $ratio_height),
1285 'height' => (int)($size[1] / $ratio_height)
1286 ];
1287 } else {
1288 return [
1289 'width' => (int)($size[0] / $ratio_width),
1290 'height' => (int)($size[1] / $ratio_width)
1291 ];
1292 }
