632 private function calculate_thumbnail_size(array $size): array
633 {
634 // Валидация параметра $size
635 if (!isset($size[0], $size[1]) || !is_int($size[0]) || !is_int($size[1]) || $size[0] <= 0 || $size[1] <= 0) {
636 throw new InvalidArgumentException(
637 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректные размеры изображения | " . "Требуется массив с двумя положительными целыми числами. Получено: width = " . (isset($size[0]) ? $size[0] : 'undefined') . ", height = " . (isset($size[1]) ? $size[1] : 'undefined')
638 );
639 }
640
641 // Валидация конфигурации
642 if (!isset($this->config['temp_photo_w'], $this->config['temp_photo_h']) || !is_numeric(
643 $this->config['temp_photo_w']
644 ) || !is_numeric(
645 $this->config['temp_photo_h']
646 ) || $this->config['temp_photo_w'] < 0 || $this->config['temp_photo_h'] < 0) {
647 throw new RuntimeException(
648 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректная конфигурация temp_photo_w или temp_photo_h | " . "Значения должны быть числами ≥ 0. Получено: temp_photo_w = " . (isset($this->config['temp_photo_w']) ? $this->config['temp_photo_w'] : 'undefined') . ", temp_photo_h = " . (isset($this->config['temp_photo_h']) ? $this->config['temp_photo_h'] : 'undefined')
649 );
650 }
651
652 // Преобразуем конфигурацию в целые числа
653 $target_width = (int)$this->config['temp_photo_w'];
654 $target_height = (int)$this->config['temp_photo_h'];
655
656 // Вычисляем коэффициенты масштабирования через match
657 $ratio_width = match (true) {
658 $target_width > 0 => $size[0] / $target_width,
659 default => 1,
660 };
661
662 $ratio_height = match (true) {
663 $target_height > 0 => $size[1] / $target_height,
664 default => 1,
665 };
666
667 // Если исходное изображение меньше целевого размера, возвращаем оригинальные размеры
668 if ($size[0] < $target_width && $size[1] < $target_height && $target_width > 0 && $target_height > 0) {
669 return [
670 'width' => $size[0],
671 'height' => $size[1]
672 ];
673 }
674
675 // Масштабируем изображение пропорционально
676 if ($ratio_width < $ratio_height) {
677 return [
678 'width' => (int)($size[0] / $ratio_height),
679 'height' => (int)($size[1] / $ratio_height)
680 ];
681 } else {
682 return [
683 'width' => (int)($size[0] / $ratio_width),
684 'height' => (int)($size[1] / $ratio_width)
685 ];
686 }

794 protected function _image_resize_internal(string $full_path, string $thumbnail_path): bool
795 {
796 // Нормализация путей через realpath()
797 $full_path = realpath($full_path);
798 $thumbnail_path = realpath($thumbnail_path);
799 if (!$full_path || !$thumbnail_path) {
800 throw new InvalidArgumentException(
801 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный путь к файлу | \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path}"
802 );
803 }
804 // Блок 1: Проверка корректности путей через filter_var
805 $path_errors = [];
806 foreach (['full_path' => $full_path, 'thumbnail_path' => $thumbnail_path] as $key => $path) {
807 if (!filter_var($path, FILTER_VALIDATE_REGEXP, [
808 'options' => ['regexp' => '/^[a-zA-Z0-9\/\.\-_]+$/']
809 ])) {
810 $path_errors[] = "Некорректный формат пути: \${$key} = {$path}.";
811 }
812 }
813 if (!empty($path_errors)) {
814 throw new InvalidArgumentException(
815 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный формат пути | Ошибки: " . implode(
816 ' ',
817 $path_errors
818 )
819 );
820 }
821 // Блок 2: Проверка существования и доступности файлов
822 if (!file_exists($full_path) || !is_readable($full_path)) {
823 throw new RuntimeException(
824 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Исходное изображение не найдено или недоступно для чтения | Путь: \$full_path = {$full_path}"
825 );
826 }
827 // Проверка доступности директории для записи
828 $thumbnail_dir = dirname($thumbnail_path);
829 if (!is_writable($thumbnail_dir)) {
830 throw new RuntimeException(
831 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Директория для сохранения эскиза недоступна для записи | Путь: \$thumbnail_dir = {$thumbnail_dir}"
832 );
833 }
834 $full_size = getimagesize($full_path);
835 if ($full_size === false) {
836 throw new RuntimeException(
837 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить размеры изображения | Путь: \$full_path = {$full_path}"
838 );
839 }
840 if ($full_size[0] > self::MAX_IMAGE_WIDTH || $full_size[1] > self::MAX_IMAGE_HEIGHT) {
841 throw new RuntimeException(
842 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размеры исходного изображения слишком велики | Путь: \$full_path = {$full_path}, ширина = {$full_size[0]}, высота = {$full_size[1]}"
843 );
844 }
845 $thumbnail_exists = file_exists($thumbnail_path);
846 if ($thumbnail_exists && !is_writable($thumbnail_path)) {
847 throw new RuntimeException(
848 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Файл эскиза существует, но недоступен для записи | Путь: \$thumbnail_path = {$thumbnail_path}"
849 );
850 }
851 // Блок 3: Расчет размеров будущего эскиза
852 $photo = $this->calculate_thumbnail_size($full_size);
853 // Если файл эскиза существует и его размеры совпадают с расчетными, завершаем работу
854 if ($thumbnail_exists) {
855 $thumbnail_size = getimagesize($thumbnail_path);
856 if ($thumbnail_size !== false && $thumbnail_size[0] == $photo['width'] && $thumbnail_size[1] == $photo['height']) {
857 return true;
858 }
859 }
860 // Блок 4: Определение MIME-типа
861 $finfo = finfo_open(FILEINFO_MIME_TYPE);
862 $mime_type = finfo_file($finfo, $full_path);
863 finfo_close($finfo);
864 // Блок 5: Подготовка данных для обработки изображения
865 $original_data = [
866 'path' => $full_path,
867 'type' => $mime_type,
868 'width' => $full_size[0],
869 'height' => $full_size[1],
870 ];
871 $thumbnail_data = [
872 'path' => $thumbnail_path,
873 'width' => $photo['width'],
874 'height' => $photo['height'],
875 'exists' => $thumbnail_exists,
876 ];
877 // Блок 6: Логика выбора библиотеки с fallback
878 try {
879 if (extension_loaded('gmagick')) {
880 return $this->process_image_resize_gmagick($original_data, $thumbnail_data);
881 }
882 } catch (Exception $e) {
883 log_in_file(
884 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через GraphicsMagick | Аргументы: \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path} | Сообщение об ошибке: {$e->getMessage()}"
885 );
886 }
887 try {
888 if (extension_loaded('imagick')) {
889 return $this->process_image_resize_imagick($original_data, $thumbnail_data);
890 }
891 } catch (Exception $e) {
892 log_in_file(
893 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через ImageMagick | Аргументы: \$full_path = {$full_path}, \$thumbnail_path = {$thumbnail_path} | Сообщение об ошибке: {$e->getMessage()}"
894 );
895 }
896 // Если ни одна из библиотек не сработала, используем GD
897 return $this->process_image_resize_gd($original_data, $thumbnail_data);
898 }

965 private function process_image_resize_gmagick(array $original_data, array $thumbnail_data): bool
966 {
967 // Разрешённые MIME-типы и их соответствие форматам GraphicsMagick
968 $gmagick_formats = [
969 'image/jpeg' => 'JPEG',
970 'image/png' => 'PNG',
971 'image/gif' => 'GIF',
972 'image/webp' => 'WEBP',

978 'image/heic' => 'HEIC', // Формат с камер iPhone
979 ];

981 if (!array_key_exists($original_data['type'], $gmagick_formats)) {
982 log_in_file(
983 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неподдерживаемый MIME-тип для GraphicsMagick | Получено: {$original_data['type']}"
984 );
985 return false;
986 }
987 // Получение формата GraphicsMagick
988 $format = $gmagick_formats[$original_data['type']];
989 // Проверка поддержки формата через GraphicsMagick
990 $gmagick_instance = new Gmagick(); // Создаем экземпляр Gmagick
991 $supported_formats = $gmagick_instance->queryFormats(); // Вызываем метод queryFormats()
992 unset($gmagick_instance);
993 if (!in_array($format, $supported_formats)) {
994 log_in_file(
995 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Формат {$format} не поддерживается GraphicsMagick"
996 );
997 return false;
998 }
999 // Ограничение памяти
1000 $available_memory = (int)ini_get('memory_limit') * 1024 * 1024; // Например, '128M' → 128 * 1024 * 1024
1001 $max_memory_usage = 0.25 * $available_memory; // 25% от доступной памяти
1002 $estimated_memory = $original_data['width'] * $original_data['height'] * 3;
1003 if ($estimated_memory > $max_memory_usage) {
1004 log_in_file(
1005 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Изображение слишком велико для обработки через GraphicsMagick | Доступная память: " . ini_get(
1006 'memory_limit'
1007 )
1008 );
1009 return false;
1010 }
1011 // Создание резервной копии старого эскиза, если он существует
1012 if ($thumbnail_data['exists']) {
1013 $backup_thumbnail_path = tempnam(dirname($thumbnail_data['path']), 'bak_');
1014 if (!$backup_thumbnail_path || !rename($thumbnail_data['path'], $backup_thumbnail_path)) {
1015 log_in_file(
1016 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось создать резервную копию старого эскиза через GraphicsMagick"
1017 );
1018 return false;
1019 }
1020 } else {
1021 $backup_thumbnail_path = null;
1022 }
1023 $image = null;
1024 try {
1025 // Создание объекта Gmagick
1026 $image = new Gmagick($original_data['path']);
1027 // Настройка прозрачности для PNG и WebP
1028 if ($original_data['type'] === 'image/png' || $original_data['type'] === 'image/webp') {
1029 $image->setImageFormat($format);
1030 $image->setImageBackgroundColor('transparent'); // Установка прозрачного фона
1031 // $image = $image->flattenImages(); // Объединение слоев с учетом прозрачности
1032 }
1033 // Масштабирование изображения
1034 $image->resizeImage(
1035 $thumbnail_data['width'],
1036 $thumbnail_data['height'],
1037 Gmagick::FILTER_LANCZOS, // Высококачественный фильтр
1038 1 // Коэффициент размытия (1 = без размытия)
1039 );
1040 // Сохранение нового эскиза
1041 $image->writeImage($thumbnail_data['path']);
1042 // Удаление резервной копии старого эскиза
1043 if ($backup_thumbnail_path && file_exists($backup_thumbnail_path)) {
1044 unlink($backup_thumbnail_path);
1045 }
1046 return true;
1047 } catch (Exception $e) {
1048 // Логирование ошибки с контекстом
1049 log_in_file(
1050 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при создании эскиза через GraphicsMagick | Оригинальный файл: {$original_data['path']}, Эскиз: {$thumbnail_data['path']}, Сообщение об ошибке: {$e->getMessage()}"
1051 );
1052 // Восстановление старого эскиза
1053 if ($backup_thumbnail_path && file_exists($backup_thumbnail_path)) {
1054 rename($backup_thumbnail_path, $thumbnail_data['path']);
1055 }
1056 return false;
1057 } finally {
1058 // Освобождение ресурсов
1059 if (isset($image) && $image instanceof Gmagick) {
1060 $image->destroy();
1061 }
1062 }
