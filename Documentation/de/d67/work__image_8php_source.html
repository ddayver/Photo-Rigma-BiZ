720 private function calculate_thumbnail_size(array $size): array
721 {
722 // Валидация параметра $size
723 if (!isset($size[0], $size[1]) || !is_int($size[0]) || !is_int($size[1]) || $size[0] <= 0 || $size[1] <= 0) {
724 throw new InvalidArgumentException(
725 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректные размеры изображения | " . "Требуется массив с двумя положительными целыми числами. Получено: width = " . ($size[0] ?? 'undefined') . ", height = " . ($size[1] ?? 'undefined')
726 );
727 }
728
729 // Валидация конфигурации
730 if (!isset($this->config['temp_photo_w'], $this->config['temp_photo_h']) || !is_numeric(
731 $this->config['temp_photo_w']
732 ) || !is_numeric(
733 $this->config['temp_photo_h']
734 ) || $this->config['temp_photo_w'] < 0 || $this->config['temp_photo_h'] < 0) {
735 throw new RuntimeException(
736 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Некорректная конфигурация temp_photo_w или temp_photo_h | " . "Значения должны быть числами ≥ 0. Получено: temp_photo_w = " . ($this->config['temp_photo_w'] ?? 'undefined') . ", temp_photo_h = " . ($this->config['temp_photo_h'] ?? 'undefined')
737 );
738 }
739
740 // Преобразуем конфигурацию в целые числа
741 $target_width = (int)$this->config['temp_photo_w'];
742 $target_height = (int)$this->config['temp_photo_h'];
743
744 // Вычисляем коэффициенты масштабирования через match
745 $ratio_width = match (true) {
746 $target_width > 0 => $size[0] / $target_width,
747 default => 1,
748 };
749
750 $ratio_height = match (true) {
751 $target_height > 0 => $size[1] / $target_height,
752 default => 1,
753 };
754
755 // Если исходное изображение меньше целевого размера, возвращаем оригинальные размеры
756 if ($size[0] < $target_width && $size[1] < $target_height && $target_width > 0 && $target_height > 0) {
757 return [
758 'width' => $size[0],
759 'height' => $size[1]
760 ];
761 }
762
763 // Масштабируем изображение пропорционально
764 if ($ratio_width < $ratio_height) {
765 return [
766 'width' => (int)($size[0] / $ratio_height),
767 'height' => (int)($size[1] / $ratio_height)
768 ];
769 }
770
771 return [
772 'width' => (int)($size[0] / $ratio_width),
773 'height' => (int)($size[1] / $ratio_width)
774 ];
775 }

905 ' ',
906 $path_errors

910 // Блок 2: Проверка существования и доступности файлов
911 if (!file_exists($full_path) || !is_readable($full_path)) {

920 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Директория для сохранения эскиза недоступна для записи | Путь: \$thumbnail_dir = $thumbnail_dir"
921 );

923 $full_size = getimagesize($full_path);
924 if ($full_size === false) {
925 throw new RuntimeException(

930 throw new RuntimeException(
931 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размеры исходного изображения слишком велики | Путь: \$full_path = $full_path, ширина = $full_size[0], высота = $full_size[1]"
932 );

938 );

942 // Если файл эскиза существует и его размеры совпадают с расчетными, завершаем работу
943 if ($thumbnail_exists) {
944 $thumbnail_size = getimagesize($thumbnail_path);
945 if ($thumbnail_size !== false && $thumbnail_size[0] === $photo['width'] && $thumbnail_size[1] === $photo['height']) {
946 return true;
947 }
948 }
949 // Блок 4: Определение MIME-типа
950 $finfo = finfo_open(FILEINFO_MIME_TYPE);
951 $mime_type = finfo_file($finfo, $full_path);
952 finfo_close($finfo);
953 // Блок 5: Подготовка данных для обработки изображения
954 $original_data = [
955 'path' => $full_path,
956 'type' => $mime_type,
957 'width' => $full_size[0],
958 'height' => $full_size[1],
959 ];
960 $thumbnail_data = [
961 'path' => $thumbnail_path,
962 'width' => $photo['width'],
963 'height' => $photo['height'],
964 'exists' => $thumbnail_exists,
965 ];
966 // Блок 6: Логика выбора библиотеки с fallback

969 return $this->process_image_resize_gmagick($original_data, $thumbnail_data);

971 } catch (Exception $e) {
972 log_in_file(
973 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через GraphicsMagick | Аргументы: \$full_path = $full_path, \$thumbnail_path = $thumbnail_path | Сообщение об ошибке: {$e->getMessage()}"
974 );
975 }
976 try {
977 if (extension_loaded('imagick')) {
978 return $this->process_image_resize_imagick($original_data, $thumbnail_data);
979 }
980 } catch (Exception $e) {
981 log_in_file(
982 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при обработке через ImageMagick | Аргументы: \$full_path = $full_path, \$thumbnail_path = $thumbnail_path | Сообщение об ошибке: {$e->getMessage()}"
983 );
984 }
985 // Если ни одна из библиотек не сработала, используем GD
986 return $this->process_image_resize_gd($original_data, $thumbnail_data);

1090 $available_memory = (int)ini_get('memory_limit') * 1024 * 1024; // Например, '128M' → 128 * 1024 * 1024
1091 $max_memory_usage = 0.25 * $available_memory; // 25% от доступной памяти
1092 $estimated_memory = $original_data['width'] * $original_data['height'] * 3;
1093 if ($estimated_memory > $max_memory_usage) {
1094 log_in_file(
1095 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Изображение слишком велико для обработки через GraphicsMagick | Доступная память: " . ini_get(
1096 'memory_limit'

1098 );

1128 1 // Коэффициент размытия (1 = без размытия)
1129 );

1131 $image->writeImage($thumbnail_data['path']);
