209function archive_old_logs(): void
210{
211 // Проверяем, подключено ли расширение zlib
212 if (!extension_loaded('zlib')) {
213 error_log(
214 date('H:i:s') .
215 " [ERROR] | UNKNOWN_IP | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Расширение zlib не подключено"
216 );
217 return;
218 }
219
220 // Проверяем существование директории логов
221 if (!is_dir(LOG_DIR) || !is_writable(LOG_DIR)) {
222 error_log(
223 date('H:i:s') .
224 " [ERROR] | " .
225 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
226 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Директория логов отсутствует или недоступна для записи | Путь: " . LOG_DIR
227 );
228 return;
229 }
230
231 // Получаем список файлов логов с расширением .txt
232 $log_files = glob(LOG_DIR . '*_log.txt');
233 if (empty($log_files)) {
234 return; // Если файлов нет, выходим
235 }
236
237 // Определяем дату неделю назад
238 $week_ago = strtotime(LOG_LIFETIME_DAYS);
239 foreach ($log_files as $file) {
240 // Извлекаем дату из имени файла
241 $file_name = basename($file);
242 if (preg_match('/^(\d{4}_\d{2}_\d{2})_log\.txt$/', $file_name, $matches)) {
243 $file_date = strtotime(str_replace('_', '-', $matches[1])); // Преобразуем дату в timestamp
244
245 // Если файл старше недели
246 if ($file_date < $week_ago) {
247 // Создаем имя архива
248 $archive_file = $file . '.gz';
249
250 // Читаем содержимое файла
251 $file_content = file_get_contents($file);
252 if ($file_content === false) {
253 error_log(
254 date('H:i:s') .
255 " [ERROR] | " .
256 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
257 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось прочитать файл | Путь: {$file}"
258 );
259 continue; // Пропускаем этот файл и переходим к следующему
260 }
261
262 // Сжимаем файл
263 $gz_handle = gzopen($archive_file, 'w' . COMPRESSION_LEVEL);
264 if ($gz_handle === false) {
265 error_log(
266 date('H:i:s') .
267 " [ERROR] | " .
268 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
269 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось создать архив | Путь: {$archive_file}"
270 );
271 continue; // Пропускаем этот файл и переходим к следующему
272 }
273
274 if (gzwrite($gz_handle, $file_content) === false) {
275 gzclose($gz_handle);
276 error_log(
277 date('H:i:s') .
278 " [ERROR] | " .
279 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
280 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось записать данные в архив | Путь: {$archive_file}"
281 );
282 continue; // Пропускаем этот файл и переходим к следующему
283 }
284 gzclose($gz_handle);
285
286 // Удаляем исходный файл
287 if (!unlink($file)) {
288 error_log(
289 date('H:i:s') .
290 " [ERROR] | " .
291 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
292 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось удалить исходный файл | Путь: {$file}"
293 );
294 }
295 }
296 }

341function log_in_file(string $txt, bool $die = false): bool
342{
343 static $log_in_process = false; // Статический флаг для предотвращения рекурсии
344 if ($log_in_process) {
345 return false;
346 }
347 $log_in_process = true;
348 try {
349 // Проверка существования директории для логов
350 if (!is_dir(LOG_DIR)) {
351 if (!mkdir(LOG_DIR, 0755, true)) {
352 throw new \RuntimeException(
353 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось создать директорию | Путь: " . LOG_DIR
354 );
355 }
356 // Проверяем права доступа после создания
357 if (!is_writable(LOG_DIR)) {
358 throw new \RuntimeException(
359 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Созданная директория недоступна для записи | Путь: " . LOG_DIR
360 );
361 }
362 }
363
364 // Проверка прав доступа к директории логов
365 if (!is_writable(LOG_DIR)) {
366 throw new \RuntimeException(
367 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Директория недоступна для записи | Путь: " . LOG_DIR
368 );
369 }
370
371 // Определение имени файла лога
372 $log_file = LOG_DIR . date('Y_m_d') . '_log.txt';
373
374 // Ограничение размера файла логов
375 if (file_exists($log_file) && filesize($log_file) > MAX_LOG_SIZE) {
376 $backup_file = $log_file . '_bak-' . date('Y_m_d_H_i_s') . '.gz';
377 if (extension_loaded('zlib')) {
378 $gz_handle = gzopen($backup_file, 'w' . COMPRESSION_LEVEL);
379 if ($gz_handle) {
380 $file_content = file_get_contents($log_file);
381 if ($file_content === false) {
382 throw new \RuntimeException(
383 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось прочитать содержимое файла | Путь: {$log_file}"
384 );
385 }
386 gzwrite($gz_handle, $file_content);
387 gzclose($gz_handle);
388 unlink($log_file); // Удаляем исходный файл после сжатия
389 }
390 } else {
391 // Меняем расширение файла с .gz на .txt
392 $backup_file = pathinfo($backup_file, PATHINFO_DIRNAME) . '/' . pathinfo($backup_file, PATHINFO_FILENAME) . '.txt';
393 rename($log_file, $backup_file); // Без сжатия
394 }
395
396 // Проверяем успешность создания архива
397 if (!file_exists($backup_file)) {
398 throw new \RuntimeException(
399 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось создать архив | Путь: {$backup_file}"
400 );
401 }
402 }
403
404 // Подготовка текста для записи в лог
405 $level = $die ? 'ERROR' : 'WARNING';
406 $remote_ip = filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP);
407 $write_txt = date('H:i:s') . " [$level] | " . ($remote_ip ?: 'UNKNOWN_IP') . ' | ' . $txt . PHP_EOL;
408
409 // Добавление трассировки, если DEBUG_GALLERY включен
410 if (defined('DEBUG_GALLERY')) {
411 $trace_depth = DEBUG_GALLERY ?? 5; // Глубина трассировки по умолчанию
412 if ($trace_depth > 0) {
413 $backtrace = debug_backtrace(DEBUG_BACKTRACE_PROVIDE_OBJECT, $trace_depth + 1);
414 array_shift($backtrace);
415 $trace_info = [];
416 foreach ($backtrace as $index => $trace) {
417 $step_number = $index + 1;
418 $args = array_map(function ($arg) {
419 $arg_str = is_string($arg) ? $arg : json_encode($arg, JSON_UNESCAPED_UNICODE);
420 return mb_strlen($arg_str, 'UTF-8') > 80
421 ? mb_substr($arg_str, 0, 80, 'UTF-8') . '...'
422 : $arg_str;
423 }, $trace['args'] ?? []);
424 $trace_info[] = sprintf(
425 "Шаг %d:" . PHP_EOL .
426 " Файл: %s" . PHP_EOL .
427 " Строка: %s" . PHP_EOL .
428 " Функция: %s" . PHP_EOL .
429 " Аргументы: %s",
430 $step_number,
431 $trace['file'] ?: 'неизвестный файл',
432 $trace['line'] ?: 'неизвестная строка',
433 $trace['function'] ?: 'неизвестная функция',
434 json_encode($args, JSON_UNESCAPED_UNICODE)
435 );
436 }
437 $write_txt .= "Трассировка:" . PHP_EOL . implode(PHP_EOL . PHP_EOL, $trace_info) . PHP_EOL;
438 }
439 }
440
441 // Безопасная запись в файл через потоки
442 $handle = fopen($log_file, 'a');
443 if ($handle) {
444 flock($handle, LOCK_EX);
445 if (fwrite($handle, $write_txt) === false) {
446 throw new \RuntimeException(
447 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось записать данные в файл | Путь: {$log_file}"
448 );
449 }
450 flock($handle, LOCK_UN);
451 fclose($handle);
452 } else {
453 throw new \RuntimeException(
454 __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Не удалось открыть файл для записи | Путь: {$log_file}"
455 );
456 }
457
458 // Завершение работы скрипта при необходимости
459 if ($die && DIE_IF_ERROR) {
460 $safe_output = htmlspecialchars($write_txt, ENT_QUOTES | ENT_HTML5, 'UTF-8');
461 die($safe_output);
462 }
463 } catch (\RuntimeException $e) {
464 error_log(
465 date('H:i:s') .
466 " [ERROR] | " .
467 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
468 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Ошибка во время выполнения | Сообщение: " . $e->getMessage()
469 );
470 $log_in_process = false;
471 return false;
472 } catch (\Exception $e) {
473 error_log(
474 date('H:i:s') .
475 " [ERROR] | " .
476 (filter_input(INPUT_SERVER, 'REMOTE_ADDR', FILTER_VALIDATE_IP) ?: 'UNKNOWN_IP') .
477 " | " . __FILE__ . ":" . __LINE__ . " (" . __FUNCTION__ . ") | Непредвиденная ошибка | Сообщение: " . $e->getMessage()
478 );
479 $log_in_process = false;
480 return false;
481 }
482 $log_in_process = false;
483 return true;
