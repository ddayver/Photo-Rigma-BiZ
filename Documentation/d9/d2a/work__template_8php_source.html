639 protected function _create_menu_internal(string $action, int $menu): array
640 {
641 // Валидация входных данных.
642 if (!in_array($menu, [0, 1], true)) {
643 throw new InvalidArgumentException(
644 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
645 );
646 }
647 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
648 $menu_type = match ($menu) {
649 0 => 'short',
650 1 => 'long',
651 default => throw new InvalidArgumentException(
652 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
653 ),
654 };
655 // Формирование запроса для получения данных меню (используем параметры для безопасности)
656 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
657 $this->db->select(
658 '*', // Все поля таблицы используются для этого запроса
659 TBL_MENU,
660 [
661 'where' => '`' . $menu_type . '` = :menu_type',
662 'params' => [':menu_type' => 1],
663 'order' => '`id` ASC',
664 ]
665 );
666 // Получение результатов запроса
667 $menu_data = $this->db->res_arr();
668 if ($menu_data === false) {
669 throw new RuntimeException(
670 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
671 );
672 }
673
674 $menu_items = [];
675 foreach ($menu_data as $key => $menu_item) {
676 $is_visible = true;
677
678 // Проверяем права доступа для текущего пользователя
679 if ($menu_item['user_login'] !== null) {
680 // user_login: 0 - только гостям, 1 - только зарегистрированным
681 if (($menu_item['user_login'] === 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
682 ($menu_item['user_login'] === 1 && $this->user->user['id'] === 0) // Зарегистрированный пытается получить доступ к пункту для гостей
683 ) {
684 $is_visible = false;
685 }
686 }
687
688 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
689 if (($menu_item['user_access'] !== null) && empty($this->user->user[$menu_item['user_access']])) {
690 $is_visible = false;
691 }
692 // Если пункт меню видим, добавляем его в результат
693 if ($is_visible) {
694 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
695 $url_action = Work::clean_field($menu_item['url_action']);
696 $name_action = Work::clean_field($menu_item['name_action']);
697 $menu_items[$key] = [
698 'url' => ($menu_item['action'] === $action ? null : $this->config['site_url'] . $url_action),
699 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action),
700 ];
701 }
702 }
703 return $menu_items;
704 }
