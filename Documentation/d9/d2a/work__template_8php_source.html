808 protected function _create_menu_internal(string $action, int $menu): array
809 {
810 // Валидация входных данных (лучше перебдеть :D)
811 if (!is_string($action)) {
812 throw new \InvalidArgumentException(
813 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип параметра \$action"
814 );
815 }
816 if (!in_array($menu, [0, 1], true)) {
817 throw new \InvalidArgumentException(
818 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
819 );
820 }
821 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
822 $menu_type = match ($menu) {
823 0 => 'short',
824 1 => 'long',
825 default => throw new \InvalidArgumentException(
826 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
827 ),
828 };
829 // Формирование запроса для получения данных меню (используем параметры для безопасности)
830 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
831 $this->db->select(
832 ['*'], // Все поля таблицы используются для этого запроса
833 TBL_MENU,
834 [
835 'where' => ':menu_type = 1',
836 'params' => [':menu_type' => $menu_type],
837 'order_by' => ['id' => 'ASC']
838 ]
839 );
840 // Получение результатов запроса
841 $menu_data = $this->db->res_arr();
842 if ($menu_data === false) {
843 throw new \RuntimeException(
844 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
845 );
846 }
847 // Формирование массива меню
848 $menu_items = [];
849 foreach ($menu_data as $key => $menu_item) {
850 $is_visible = true;
851 // Проверяем права доступа для текущего пользователя
852 // user_login: 0 - гость, 1 - не гость.
853 // user_access: 0 - нельзя смотреть, 1 - можно смотреть.
854 if (($menu_item['user_login'] !== '') && (
855 ($menu_item['user_login'] == 0 && $this->user->user['id'] > 0) ||
856 ($menu_item['user_login'] == 1 && $this->user->user['id'] == 0)
857 )) {
858 $is_visible = false;
859 }
860 if ($menu_item['user_access'] !== '' && $this->user->user[$menu_item['user_access']] != 1) {
861 $is_visible = false;
862 }
863 // Если пункт меню видим, добавляем его в результат
864 if ($is_visible) {
865 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
866 $url_action = Work::clean_field($menu_item['url_action']);
867 $name_action = Work::clean_field($menu_item['name_action']);
868 $menu_items[$key] = [
869 'url' => ($menu_item['action'] == $action ? null : $this->config['site_url'] . $url_action),
870 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action)
871 ];
872 }
873 }
874 return $menu_items;
875 }
