611 protected function _create_menu_internal(string $action, int $menu): array
612 {
613 // Валидация входных данных (лучше перебдеть :D)
614 if (!is_string($action)) {
615 throw new \InvalidArgumentException(
616 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип параметра \$action"
617 );
618 }
619 if (!in_array($menu, [0, 1], true)) {
620 throw new \InvalidArgumentException(
621 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
622 );
623 }
624 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
625 $menu_type = match ($menu) {
626 0 => 'short',
627 1 => 'long',
628 default => throw new \InvalidArgumentException(
629 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
630 ),
631 };
632 // Формирование запроса для получения данных меню (используем параметры для безопасности)
633 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
634 $this->db->select(
635 ['*'], // Все поля таблицы используются для этого запроса
636 TBL_MENU,
637 [
638 'where' => ':menu_type = 1',
639 'params' => [':menu_type' => $menu_type],
640 'order_by' => ['id' => 'ASC']
641 ]
642 );
643 // Получение результатов запроса
644 $menu_data = $this->db->res_arr();
645 if ($menu_data === false) {
646 throw new \RuntimeException(
647 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
648 );
649 }
650 // Формирование массива меню
651 $menu_items = [];
652 foreach ($menu_data as $key => $menu_item) {
653 $is_visible = true;
654 // Проверяем права доступа для текущего пользователя
655 // user_login: 0 - гость, 1 - не гость.
656 // user_access: 0 - нельзя смотреть, 1 - можно смотреть.
657 if (($menu_item['user_login'] !== '') && (
658 ($menu_item['user_login'] == 0 && $this->user->user['id'] > 0) ||
659 ($menu_item['user_login'] == 1 && $this->user->user['id'] == 0)
660 )) {
661 $is_visible = false;
662 }
663 if ($menu_item['user_access'] !== '' && $this->user->user[$menu_item['user_access']] != 1) {
664 $is_visible = false;
665 }
666 // Если пункт меню видим, добавляем его в результат
667 if ($is_visible) {
668 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
669 $url_action = Work::clean_field($menu_item['url_action']);
670 $name_action = Work::clean_field($menu_item['name_action']);
671 $menu_items[$key] = [
672 'url' => ($menu_item['action'] == $action ? null : $this->config['site_url'] . $url_action),
673 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action)
674 ];
675 }
676 }
677 return $menu_items;
678 }
