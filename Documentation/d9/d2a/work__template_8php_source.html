651 protected function _create_menu_internal(string $action, int $menu): array
652 {
653 // Валидация входных данных.
654 if (!in_array($menu, [0, 1], true)) {
655 throw new InvalidArgumentException(
656 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
657 );
658 }
659 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
660 $menu_type = match ($menu) {
661 0 => 'short',
662 1 => 'long',
663 default => throw new InvalidArgumentException(
664 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
665 ),
666 };
667 // Формирование запроса для получения данных меню (используем параметры для безопасности)
668 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
669 $this->db->select(
670 '*', // Все поля таблицы используются для этого запроса
671 TBL_MENU,
672 [
673 'where' => '`' . $menu_type . '` = :menu_type',
674 'params' => [':menu_type' => 1],
675 'order' => '`id` ASC',
676 ]
677 );
678 // Получение результатов запроса
679 $menu_data = $this->db->res_arr();
680 if ($menu_data === false) {
681 throw new RuntimeException(
682 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
683 );
684 }
685
686 $menu_items = [];
687 foreach ($menu_data as $key => $menu_item) {
688 $is_visible = true;
689
690 // Проверяем права доступа для текущего пользователя
691 if ($menu_item['user_login'] !== null) {
692 // user_login: 0 - только гостям, 1 - только зарегистрированным
693 if (($menu_item['user_login'] === 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
694 ($menu_item['user_login'] === 1 && $this->user->user['id'] === 0) // Зарегистрированный пытается получить доступ к пункту для гостей
695 ) {
696 $is_visible = false;
697 }
698 }
699
700 if ($menu_item['user_access'] !== null) {
701 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
702 if (empty($this->user->user[$menu_item['user_access']])) {
703 $is_visible = false;
704 }
705 }
706 // Если пункт меню видим, добавляем его в результат
707 if ($is_visible) {
708 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
709 $url_action = Work::clean_field($menu_item['url_action']);
710 $name_action = Work::clean_field($menu_item['name_action']);
711 $menu_items[$key] = [
712 'url' => ($menu_item['action'] === $action ? null : $this->config['site_url'] . $url_action),
713 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action),
714 ];
715 }
716 }
717 return $menu_items;
718 }
