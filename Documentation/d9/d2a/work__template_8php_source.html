641 protected function _create_menu_internal(string $action, int $menu): array
642 {
643 // Валидация входных данных.
644 if (!in_array($menu, [0, 1], true)) {
645 throw new InvalidArgumentException(
646 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
647 );
648 }
649 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
650 $menu_type = match ($menu) {
651 0 => 'short',
652 1 => 'long',
653 default => throw new InvalidArgumentException(
654 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
655 ),
656 };
657 // Формирование запроса для получения данных меню (используем параметры для безопасности)
658 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
659 $this->db->select(
660 ['*'], // Все поля таблицы используются для этого запроса
661 TBL_MENU,
662 [
663 'where' => '`' . $menu_type . '` = :menu_type',
664 'params' => [':menu_type' => 1],
665 'order_by' => ['id' => 'ASC']
666 ]
667 );
668 // Получение результатов запроса
669 $menu_data = $this->db->res_arr();
670 if ($menu_data === false) {
671 throw new RuntimeException(
672 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
673 );
674 }
675
676 $menu_items = [];
677 foreach ($menu_data as $key => $menu_item) {
678 $is_visible = true;
679
680 // Проверяем права доступа для текущего пользователя
681 if ($menu_item['user_login'] !== null) {
682 // user_login: 0 - только гостям, 1 - только зарегистрированным
683 if (($menu_item['user_login'] === 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
684 ($menu_item['user_login'] === 1 && $this->user->user['id'] === 0) // Зарегистрированный пытается получить доступ к пункту для гостей
685 ) {
686 $is_visible = false;
687 }
688 }
689
690 if ($menu_item['user_access'] !== null) {
691 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
692 if (empty($this->user->user[$menu_item['user_access']])) {
693 $is_visible = false;
694 }
695 }
696 // Если пункт меню видим, добавляем его в результат
697 if ($is_visible) {
698 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
699 $url_action = Work::clean_field($menu_item['url_action']);
700 $name_action = Work::clean_field($menu_item['name_action']);
701 $menu_items[$key] = [
702 'url' => ($menu_item['action'] === $action ? null : $this->config['site_url'] . $url_action),
703 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action)
704 ];
705 }
706 }
707 return $menu_items;
708 }
