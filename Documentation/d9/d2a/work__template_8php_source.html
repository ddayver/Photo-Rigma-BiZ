795 protected function _create_menu_internal(string $action, int $menu): array
796 {
797 // Валидация входных данных (лучше перебдеть :D)
798 if (!is_string($action)) {
799 throw new \InvalidArgumentException(
800 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип параметра \$action"
801 );
802 }
803 if (!in_array($menu, [0, 1], true)) {
804 throw new \InvalidArgumentException(
805 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
806 );
807 }
808 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
809 $menu_type = match ($menu) {
810 0 => 'short',
811 1 => 'long',
812 default => throw new \InvalidArgumentException(
813 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
814 ),
815 };
816 // Формирование запроса для получения данных меню (используем параметры для безопасности)
817 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
818 $this->db->select(
819 ['*'], // Все поля таблицы используются для этого запроса
820 TBL_MENU,
821 [
822 'where' => '`' . $menu_type. '` = :menu_type',
823 'params' => [':menu_type' => 1],
824 'order_by' => ['id' => 'ASC']
825 ]
826 );
827 // Получение результатов запроса
828 $menu_data = $this->db->res_arr();
829 if ($menu_data === false) {
830 throw new \RuntimeException(
831 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
832 );
833 }
834
835 $menu_items = [];
836 foreach ($menu_data as $key => $menu_item) {
837 $is_visible = true;
838
839 // Проверяем права доступа для текущего пользователя
840 if ($menu_item['user_login'] !== null) {
841 // user_login: 0 - только гостям, 1 - только зарегистрированным
842 if (
843 ($menu_item['user_login'] == 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
844 ($menu_item['user_login'] == 1 && $this->user->user['id'] == 0) // Зарегистрированный пытается получить доступ к пункту для гостей
845 ) {
846 $is_visible = false;
847 }
848 }
849
850 if ($menu_item['user_access'] !== null) {
851 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
852 if (!isset($this->user->user[$menu_item['user_access']]) || $this->user->user[$menu_item['user_access']] != 1) {
853 $is_visible = false;
854 }
855 }
856
857 // Если пункт меню видим, добавляем его в результат
858 if ($is_visible) {
859 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
860 $url_action = Work::clean_field($menu_item['url_action']);
861 $name_action = Work::clean_field($menu_item['name_action']);
862 $menu_items[$key] = [
863 'url' => ($menu_item['action'] == $action ? null : $this->config['site_url'] . $url_action),
864 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action)
865 ];
866 }
867 }
868 return $menu_items;
869 }
