816 protected function _create_menu_internal(string $action, int $menu): array
817 {
818 // Валидация входных данных (лучше перебдеть :D)
819 if (!is_string($action)) {
820 throw new \InvalidArgumentException(
821 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип параметра \$action"
822 );
823 }
824 if (!in_array($menu, [0, 1], true)) {
825 throw new \InvalidArgumentException(
826 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
827 );
828 }
829 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
830 $menu_type = match ($menu) {
831 0 => 'short',
832 1 => 'long',
833 default => throw new \InvalidArgumentException(
834 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
835 ),
836 };
837 // Формирование запроса для получения данных меню (используем параметры для безопасности)
838 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
839 $this->db->select(
840 ['*'], // Все поля таблицы используются для этого запроса
841 TBL_MENU,
842 [
843 'where' => ':menu_type = 1',
844 'params' => [':menu_type' => $menu_type],
845 'order_by' => ['id' => 'ASC']
846 ]
847 );
848 // Получение результатов запроса
849 $menu_data = $this->db->res_arr();
850 if ($menu_data === false) {
851 throw new \RuntimeException(
852 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
853 );
854 }
855 // Формирование массива меню
856 $menu_items = [];
857 foreach ($menu_data as $key => $menu_item) {
858 $is_visible = true;
859 // Проверяем права доступа для текущего пользователя
860 // user_login: 0 - гость, 1 - не гость.
861 // user_access: 0 - нельзя смотреть, 1 - можно смотреть.
862 if (($menu_item['user_login'] !== '') && (
863 ($menu_item['user_login'] == 0 && $this->user->user['id'] > 0) ||
864 ($menu_item['user_login'] == 1 && $this->user->user['id'] == 0)
865 )) {
866 $is_visible = false;
867 }
868 if ($menu_item['user_access'] !== '' && $this->user->user[$menu_item['user_access']] != 1) {
869 $is_visible = false;
870 }
871 // Если пункт меню видим, добавляем его в результат
872 if ($is_visible) {
873 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
874 $url_action = Work::clean_field($menu_item['url_action']);
875 $name_action = Work::clean_field($menu_item['name_action']);
876 $menu_items[$key] = [
877 'url' => ($menu_item['action'] == $action ? null : $this->config['site_url'] . $url_action),
878 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action)
879 ];
880 }
881 }
882 return $menu_items;
883 }
