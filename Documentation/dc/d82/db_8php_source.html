1071 private function sanitize_expression(string $expression): string
1072 {
1073 return $expression; // Временная заглушка
1074 // Логируем входное выражение для отладки
1075 error_log("Полученное выражение в sanitize_expression: $expression");
1076
1077 // === 1. Валидация аргументов ===
1078 if (!is_string($expression)) {
1079 throw new InvalidArgumentException(
1080 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверный тип выражения | Ожидалась строка, получено: " . gettype(
1081 $expression
1082 )
1083 );
1084 }
1085
1086 // === 2. Проверка на наличие сложных SQL-функций ===
1087 if (preg_match('/^[A-Z_]+\‍(/i', $expression)) {
1088 // Улучшенное регулярное выражение для функций
1089 if (!preg_match('/^[A-Z_]+\‍((?:[^()]*(?:\‍([^()]*\‍)[^()]*)*)\‍)(\s+AS\s+[a-zA-Z0-9_]+)?$/i', $expression)) {
1090 throw new InvalidArgumentException(
1091 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимая SQL-функция | Получено: $expression"
1092 );
1093 }
1094 return $expression; // Возвращаем выражение без изменений
1095 }
1096
1097 // === 3. Обработка условий с операторами ===
1098 if (preg_match(
1099 '/^(`?[a-zA-Z0-9_\.]+`?)\s*(=|<|>|<=|>=|!=|<>|LIKE|IN|IS NULL)\s*(:[a-zA-Z0-9_]+|\'[^\']*\'|"[^"]*"|\d+)/i',
1100 $expression
1101 )) {
1102 // Разбиваем условие на части
1103 $pattern = '/^(`?[a-zA-Z0-9_\.]+`?)\s*(=|<|>|<=|>=|!=|<>|LIKE|IN|IS NULL)\s*(:[a-zA-Z0-9_]+|\'[^\']*\'|"[^"]*"|\d+)/i';
1104 if (preg_match($pattern, $expression, $matches)) {
1105 $field = $matches[1]; // Поле (например, `id`)
1106 $operator = $matches[2]; // Оператор (например, `=` или `!=`)
1107 $value = $matches[3]; // Значение (например, `:group_id`)
1108
1109 // Экранируем поле, если оно не экранировано
1110 if (!preg_match('/^`.*`$/', $field)) {
1111 if (strpos($field, '.') !== false) {
1112 list($table, $column) = explode('.', $field, 2);
1113 $field = "`$table`.`$column`";
1114 } else {
1115 $field = "`$field`";
1116 }
1117 }
1118
1119 // Возвращаем условие в исходном виде
1120 return "$field $operator $value";
1121 }
1122 }
1123
1124 // === 4. Обработка остальных частей ===
1125 // Если часть — это символ *, возвращаем его без изменений
1126 if ($expression === '*') {
1127 return $expression;
1128 }
1129
1130 // Если часть — это имя таблицы или поля, экранируем её
1131 if (preg_match('/^(`?[a-zA-Z0-9_\.]+`?)$/', $expression)) {
1132 // Если имя уже экранировано, возвращаем его без изменений
1133 if (preg_match('/^`.*`$/', $expression)) {
1134 return $expression;
1135 }
1136 // Если имя не экранировано, добавляем обратные кавычки
1137 if (strpos($expression, '.') !== false) {
1138 list($table, $column) = explode('.', $expression, 2);
1139 return "`$table`.`$column`";
1140 }
1141 return "`$expression`";
1142 }
1143
1144 // Если часть — это значение в кавычках, возвращаем её без изменений
1145 if (preg_match('/^([\'"]).*\1$/', $expression)) {
1146 return $expression;
1147 }
1148
1149 // Если часть — это параметр SQL-запроса (начинается с ':'), возвращаем его без изменений
1150 if (preg_match('/^:[a-zA-Z0-9_]+$/', $expression)) {
1151 return $expression;
1152 }
1153
1154 // Если часть — это подзапрос, рекурсивно обрабатываем его
1155 if (preg_match('/^\‍(.*\‍)$/', $expression)) {
1156 $subquery = substr($expression, 1, -1);
1157 $sanitized_subquery = $this->sanitize_expression($subquery);
1158 return "($sanitized_subquery)";
1159 }
1160
1161 // Если часть — это что-то другое, выбрасываем исключение
1162 throw new InvalidArgumentException(
1163 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимая часть выражения | Получено: $expression"
1164 );
