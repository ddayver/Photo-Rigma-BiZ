1551 private function log_explain(array $params): void
1552 {
1553 $explain = '';
1554 try {
1555 // Определяем, нужно ли добавлять ANALYZE
1556 $is_write_query = stripos(trim($this->txt_query), 'INSERT') === 0 || stripos(
1557 trim($this->txt_query),
1558 'UPDATE'
1559 ) === 0 || stripos(trim($this->txt_query), 'DELETE') === 0 || stripos(
1560 trim($this->txt_query),
1561 'TRUNCATE'
1562 ) === 0;
1563
1564 // Определяем тип EXPLAIN в зависимости от СУБД и версии
1565 if ($this->db_type === 'sqlite') {
1566 $explain = 'EXPLAIN'; // Для SQLite всегда только EXPLAIN
1567 } elseif ($this->db_type === 'mysql') {
1568 // Для MySQL или MariaDB проверяем версию и тип СУБД
1569 try {
1570 $stmt = $this->pdo->query("SELECT VERSION()");
1571 $version = $stmt->fetchColumn();
1572
1573 // Проверяем, является ли СУБД MariaDB
1574 $is_mariadb = stripos($version, 'mariadb') !== false;
1575
1576 // Извлекаем числовую часть версии (например, "8.0.30" -> 8.0)
1577 preg_match('/^(\d+\.\d+)/', $version, $matches);
1578 $numeric_version = $matches[1] ?? null;
1579
1580 // Определяем, поддерживается ли ANALYZE
1581 $supports_analyze = !$is_mariadb && $numeric_version !== null && version_compare(
1582 $numeric_version,
1583 '5.7',
1584 '>='
1585 );
1586
1587 $explain = $is_write_query || !$supports_analyze ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE или для MariaDB/MySQL < 5.7
1588 : 'EXPLAIN ANALYZE';
1589 } catch (Exception) {
1590 // Если не удалось получить версию, используем только EXPLAIN
1591 $explain = 'EXPLAIN';
1592 }
1593 } elseif ($this->db_type === 'pgsql') {
1594 $explain = $is_write_query ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE
1595 : 'EXPLAIN ANALYZE'; // С ANALYZE для SELECT
1596 } else {
1597 throw new InvalidArgumentException(
1598 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не поддерживаемый тип СУБД | Тип: $this->db_type"
1599 );
1600 }
1601
1602 // Выполняем EXPLAIN (или EXPLAIN ANALYZE)
1603 $stmt = $this->pdo->prepare("$explain $this->txt_query");
1604 $stmt->execute($params);
1605
1606 // Получаем результат
1607 $explain_rows = $stmt->fetchAll(PDO::FETCH_ASSOC);
1608
1609 // Форматируем результат для логирования
1610 $explain_result = '';
1611 foreach ($explain_rows as $row) {
1612 $explain_result .= implode(
1613 ' | ',
1614 array_map(static fn($key, $value) => "$key: $value", array_keys($row), $row)

1616 }

1621 $params,
1622 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1623 ) : "отсутствуют") . PHP_EOL . "$explain Результат:" . PHP_EOL . $explain_result
1624 );
1625 } catch (Exception $e) {
1626 // Логируем ошибку
1627 log_in_file(
1628 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при выполнении $explain: | " . "Запрос: " . $this->txt_query . PHP_EOL . "Параметры: " . (!empty($params) ? json_encode(
1629 $params,
1630 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1631 ) : "отсутствуют") . PHP_EOL . "Сообщение об ошибке: {$e->getMessage()}"
1632 );

1675 private function log_query(string $reason = 'other', ?float $execution_time_ms = null): void
1676 {
1677 // Проверяем, что запрос существует, является строкой и не является запросом к таблице логов
1678 if (!is_string($this->txt_query) || empty($this->txt_query) || str_contains($this->txt_query, TBL_LOG_QUERY)) {
1679 return;
1680 }
1681
1682 // Проверяем и нормализуем время выполнения (минимальное значение 0)
1683 $execution_time_ms = max(0.0, (float)($execution_time_ms ?? 0));
1684
1685 // Обрезаем запрос до допустимой длины (например, 65,530 символа для TEXT в MySQL)
1686 $max_query_length = 65530; // Максимальная длина для TEXT
1687 $log_txt_query = strlen($this->txt_query) > $max_query_length ? substr(
1688 $this->txt_query,
1689 0,
1690 $max_query_length
1691 ) . '...' // Добавляем суффикс, если текст обрезан
1692 : $this->txt_query;
1693
1694 // Проверяем значение $reason через match
1695 $reason = match ($reason) {
1696 'slow', 'no_placeholders' => $reason, // Оставляем только допустимые значения
1697 default => 'other',
1698 };
1699
1700 // Хэшируем запрос для проверки на дублирование
1701 $hash = md5($this->txt_query);
1702
1703 // Проверяем кэш на наличие запроса
1704 if (!isset($this->query_cache[$hash])) {
1705 // Если запроса нет в кэше, проверяем его наличие в базе данных
1706 $stmt = $this->pdo->prepare(
1707 "SELECT id, usage_count, execution_time FROM " . TBL_LOG_QUERY . " WHERE query_hash = :hash"
1708 );
1709 $stmt->execute([':hash' => $hash]);
1710 $this->query_cache[$hash] = $stmt->fetch(PDO::FETCH_ASSOC); // Сохраняем результат в кэше
1711 }
1712
