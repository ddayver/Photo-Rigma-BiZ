1052 private function sanitize_expression(string $expression): string
1053 {
1054 return $expression; // Временная заглушка
1055 // Логируем входное выражение для отладки
1056 error_log("Полученное выражение в sanitize_expression: $expression");
1057
1058 // === 1. Валидация аргументов ===
1059 if (!is_string($expression)) {
1060 throw new InvalidArgumentException(
1061 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверный тип выражения | Ожидалась строка, получено: " . gettype(
1062 $expression
1063 )
1064 );
1065 }
1066
1067 // === 2. Проверка на наличие сложных SQL-функций ===
1068 if (preg_match('/^[A-Z_]+\‍(/i', $expression)) {
1069 // Улучшенное регулярное выражение для функций
1070 if (!preg_match('/^[A-Z_]+\‍((?:[^()]*(?:\‍([^()]*\‍)[^()]*)*)\‍)(\s+AS\s+\w+)?$/i', $expression)) {
1071 throw new InvalidArgumentException(
1072 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимая SQL-функция | Получено: $expression"
1073 );
1074 }
1075 return $expression; // Возвращаем выражение без изменений
1076 }
1077
1078 // === 3. Обработка условий с операторами ===
1079 if (preg_match(
1080 '/^(`?[a-zA-Z0-9_\.]+`?)\s*(=|<|>|<=|>=|!=|<>|LIKE|IN|IS NULL)\s*(:\w+|\'[^\']*\'|"[^"]*"|\d+)/i',
1081 $expression
1082 )) {
1083 // Разбиваем условие на части
1084 $pattern = '/^(`?[a-zA-Z0-9_\.]+`?)\s*(=|<|>|<=|>=|!=|<>|LIKE|IN|IS NULL)\s*(:\w+|\'[^\']*\'|"[^"]*"|\d+)/i';
1085 if (preg_match($pattern, $expression, $matches)) {
1086 [$field, $operator, $value] = array_slice($matches, 1, 3);
1087
1088 // Экранируем поле, если оно не экранировано
1089 if (!preg_match('/^`.*`$/', $field)) {
1090 if (str_contains($field, '.')) {
1091 [$table, $column] = explode('.', $field, 2);
1092 $field = "`$table`.`$column`";
1093 } else {
1094 $field = "`$field`";
1095 }
1096 }
1097
1098 // Возвращаем условие в исходном виде
1099 return "$field $operator $value";
1100 }
1101 }
1102
1103 // === 4. Обработка остальных частей ===
1104 // Если часть — это символ *, возвращаем его без изменений
1105 if ($expression === '*') {
1106 return $expression;
1107 }
1108
1109 // Если часть — это имя таблицы или поля, экранируем её
1110 if (preg_match('/^(`?[a-zA-Z0-9_\.]+`?)$/', $expression)) {
1111 // Если имя уже экранировано, возвращаем его без изменений
1112 if (preg_match('/^`.*`$/', $expression)) {
1113 return $expression;
1114 }
1115 // Если имя не экранировано, добавляем обратные кавычки

1118 return "`$table`.`$column`";

1120 return "`$expression`";
1121 }

1125 return $expression;
1126 }

1130 return $expression;
1131 }

1200 {

1205 );
1206 }

1215 $placeholder = $value;
