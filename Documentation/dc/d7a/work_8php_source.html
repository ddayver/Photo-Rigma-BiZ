276 {
277 if ($name === 'config') {

1237 private function validate_array(array $array, int $max_depth, int $current_depth = 1): array
1238 {
1239 $errors = [];
1240 foreach ($array as $key => $value) {
1241 if (!is_string($key)) {
1242 $errors[] = "Ключ '{$key}' на уровне {$current_depth} должен быть строкой.";
1243 }
1244 if (is_array($value)) {
1245 if ($current_depth >= $max_depth) {
1246 $errors[] = "Глубина массива превышает допустимое значение ({$max_depth}).";
1247 } else {
1248 $nested_errors = $this->validate_array($value, $max_depth, $current_depth + 1);
1249 $errors = array_merge($errors, $nested_errors);
1250 }
1251 } elseif (!is_string($value)) {
1252 $errors[] = "Значение для ключа '{$key}' на уровне {$current_depth} должно быть строкой.";
