1941 private function validate_array(array $array, int $max_depth, int $current_depth = 1): array
1942 {
1943 $errors = [];
1944 foreach ($array as $key => $value) {
1945 if (!is_string($key)) {
1946 $errors[] = "Ключ '{$key}' на уровне {$current_depth} должен быть строкой.";
1947 }
1948 if (is_array($value)) {
1949 if ($current_depth >= $max_depth) {
1950 $errors[] = "Глубина массива превышает допустимое значение ({$max_depth}).";
1951 } else {
1952 $nested_errors = $this->validate_array($value, $max_depth, $current_depth + 1);
1953 $errors = array_merge($errors, $nested_errors);
1954 }
1955 } elseif (!is_string($value)) {
1956 $errors[] = "Значение для ключа '{$key}' на уровне {$current_depth} должно быть строкой.";
1957 }
1958 }
1959 return $errors;
1960 }
