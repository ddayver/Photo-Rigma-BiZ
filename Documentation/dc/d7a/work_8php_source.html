771 private function process_lang_array(array $lang): array
772 {
773 $result = [
774 'errors' => [],
775 'changes' => [],
776 'result' => $lang, // Инициализируем результат исходным массивом
777 ];
778
779 foreach ($lang as $name => $keys) {
780 // Проверка ключей первого уровня
781 if (!is_string($name)) {
782 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
783 continue;
784 }
785
786 // Проверка второго уровня
787 if (!is_array($keys)) {
788 $result['errors'][] = "Значение для ключа '{$name}' должно быть массивом.";
789 continue;
790 }
791
792 foreach ($keys as $key => $value) {
793 // Преобразование ключа в строку
794 $test_key = (string)$key;
795 if (empty($test_key)) {
796 $result['errors'][] = "Ключ второго уровня для '{$name}' должен быть строкой. Получено: {$key} - " . gettype(
797 $key
798 );
799 continue;
800 }
801
802 // Удаление старого ключа и добавление нового
803 unset($result['result'][$name][$key]);
804 $key = $test_key;
805
806 // Проверка значений второго уровня
807 if (!is_string($value)) {
808 $result['errors'][] = "Значение для ключа '{$name}['{$key}']' должно быть строкой. Получено: " . gettype(
809 $value
810 );
811 continue;
812 }
813
814 // Очистка значения
815 $original_value = $value;
816 $cleaned_value = self::clean_field($value);
817 if ($original_value !== $cleaned_value) {
818 $result['changes']["{$name}['{$key}']"] = [
819 'original' => $original_value,
820 'cleaned' => $cleaned_value,
821 ];
822 $value = $cleaned_value; // Обновляем значение
823 }
824
825 // Добавляем обработанный ключ и значение в результирующий массив
826 $result['result'][$name][$key] = $value;
