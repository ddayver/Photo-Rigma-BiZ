775 private function process_lang_array(array $lang): array
776 {
777 $result = [
778 'errors' => [],
779 'changes' => [],
780 'result' => $lang, // Инициализируем результат исходным массивом
781 ];
782
783 foreach ($lang as $name => $keys) {
784 // Проверка ключей первого уровня
785 if (!is_string($name)) {
786 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
787 continue;
788 }
789
790 // Проверка второго уровня
791 if (!is_array($keys)) {
792 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
793 continue;
794 }
795
796 foreach ($keys as $key => $value) {
797 // Преобразование ключа в строку
798 $test_key = (string)$key;
799 if (empty($test_key)) {
800 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
801 $key
802 );
803 continue;
804 }
805
806 // Удаление старого ключа и добавление нового
807 unset($result['result'][$name][$key]);
808 $key = $test_key;
809
810 // Проверка значений второго уровня
811 if (!is_string($value)) {
812 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
813 $value
814 );
815 continue;
816 }
817
818 // Очистка значения
819 $original_value = $value;
820 $cleaned_value = self::clean_field($value);
821 if ($original_value !== $cleaned_value) {
822 $result['changes']["{$name}['$key']"] = [
823 'original' => $original_value,
824 'cleaned' => $cleaned_value,
825 ];
826 $value = $cleaned_value; // Обновляем значение
827 }
828
829 // Добавляем обработанный ключ и значение в результирующий массив
830 $result['result'][$name][$key] = $value;
831 }
832 }
833
834 return $result;
