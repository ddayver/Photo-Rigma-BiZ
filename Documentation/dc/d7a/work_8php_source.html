756 private function process_lang_array(array $lang): array
757 {
758 $result = [
759 'errors' => [],
760 'changes' => [],
761 'result' => $lang, // Инициализируем результат исходным массивом
762 ];
763
764 foreach ($lang as $name => $keys) {
765 // Проверка ключей первого уровня
766 if (!is_string($name)) {
767 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
768 continue;
769 }
770
771 // Проверка второго уровня
772 if (!is_array($keys)) {
773 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
774 continue;
775 }
776
777 foreach ($keys as $key => $value) {
778 // Преобразование ключа в строку
779 $test_key = (string)$key;
780 if (empty($test_key)) {
781 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
782 $key
783 );
784 continue;
785 }
786
787 // Удаление старого ключа и добавление нового
788 unset($result['result'][$name][$key]);
789 $key = $test_key;
790
791 // Проверка значений второго уровня
792 if (!is_string($value)) {
793 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
794 $value
795 );
796 continue;
797 }
798
799 // Очистка значения
800 $original_value = $value;
801 $cleaned_value = self::clean_field($value);
802 if ($original_value !== $cleaned_value) {
803 $result['changes']["{$name}['$key']"] = [
804 'original' => $original_value,
805 'cleaned' => $cleaned_value,
806 ];
807 $value = $cleaned_value; // Обновляем значение
808 }
809
810 // Добавляем обработанный ключ и значение в результирующий массив
811 $result['result'][$name][$key] = $value;
812 }
813 }
814
815 return $result;
816 }
