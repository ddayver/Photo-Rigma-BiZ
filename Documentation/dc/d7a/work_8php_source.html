761 private function process_lang_array(array $lang): array
762 {
763 $result = [
764 'errors' => [],
765 'changes' => [],
766 'result' => $lang, // Инициализируем результат исходным массивом
767 ];
768
769 foreach ($lang as $name => $keys) {
770 // Проверка ключей первого уровня
771 if (!is_string($name)) {
772 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
773 continue;
774 }
775
776 // Проверка второго уровня
777 if (!is_array($keys)) {
778 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
779 continue;
780 }
781
782 foreach ($keys as $key => $value) {
783 // Преобразование ключа в строку
784 $test_key = (string)$key;
785 if (empty($test_key)) {
786 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
787 $key
788 );
789 continue;
790 }
791
792 // Удаление старого ключа и добавление нового
793 unset($result['result'][$name][$key]);
794 $key = $test_key;
795
796 // Проверка значений второго уровня
797 if (!is_string($value)) {
798 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
799 $value
800 );
801 continue;
802 }
803
804 // Очистка значения
805 $original_value = $value;
806 $cleaned_value = self::clean_field($value);
807 if ($original_value !== $cleaned_value) {
808 $result['changes']["{$name}['$key']"] = [
809 'original' => $original_value,
810 'cleaned' => $cleaned_value,
811 ];
812 $value = $cleaned_value; // Обновляем значение
813 }
814
815 // Добавляем обработанный ключ и значение в результирующий массив
816 $result['result'][$name][$key] = $value;
817 }
818 }
819
820 return $result;
