785 private function process_lang_array(array $lang): array
786 {
787 $result = [
788 'errors' => [],
789 'changes' => [],
790 'result' => $lang, // Инициализируем результат исходным массивом
791 ];
792
793 foreach ($lang as $name => $keys) {
794 // Проверка ключей первого уровня
795 if (!is_string($name)) {
796 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
797 continue;
798 }
799
800 // Проверка второго уровня
801 if (!is_array($keys)) {
802 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
803 continue;
804 }
805
806 foreach ($keys as $key => $value) {
807 // Преобразование ключа в строку
808 $test_key = (string)$key;
809 if (empty($test_key)) {
810 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
811 $key
812 );
813 continue;
814 }
815
816 // Удаление старого ключа и добавление нового
817 unset($result['result'][$name][$key]);
818 $key = $test_key;
819
820 // Проверка значений второго уровня
821 if (!is_string($value)) {
822 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
823 $value
824 );
825 continue;
826 }
827
828 // Очистка значения
829 $original_value = $value;
830 $cleaned_value = self::clean_field($value);
831 if ($original_value !== $cleaned_value) {
832 $result['changes']["{$name}['$key']"] = [
833 'original' => $original_value,
834 'cleaned' => $cleaned_value,
835 ];
836 $value = $cleaned_value; // Обновляем значение
837 }
838
839 // Добавляем обработанный ключ и значение в результирующий массив
840 $result['result'][$name][$key] = $value;
841 }
842 }
843
844 return $result;
845 }
