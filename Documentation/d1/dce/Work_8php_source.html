314 private function load_cached_config(Database_Interface $db): array
315 {
316 // Получаем время последнего изменения таблицы TBL_CONFIG из таблицы TBL_CHANGE_TIMESTAMP
317 $table_name = str_replace('`', '', TBL_CONFIG); // Убираем символы экранирования
318 $db->select('last_update', TBL_CHANGE_TIMESTAMP, [
319 'where' => '`table_name` = :table_name',
320 'params' => [':table_name' => $table_name],
321 ]);
322 $timestamp_result = $db->res_row();
323
324 // Если время последнего изменения таблицы не найдено, выбрасываем исключение
325 if (!$timestamp_result || !isset($timestamp_result['last_update'])) {
326 throw new RuntimeException(
327 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить дату последнего изменения таблицы: $table_name"
328 );
329 }
330
331 // Преобразуем дату последнего изменения в timestamp для использования в качестве контрольной суммы
332 $db_last_update = strtotime($timestamp_result['last_update']);
333
334 // Проверяем актуальность кеша
335 $config_data = $this->cache->is_valid('config_cache', $db_last_update);
336 if (!$config_data) {
337 // Если кеш устарел или отсутствует, загружаем данные из БД
338 $db->select('*', TBL_CONFIG, ['params' => []]);
339 $result = $db->res_arr();
340
341 // Загружаем все данные из таблицы конфигурации
342 if ($result === false) {

349 $config_data = array_column($result, 'value', 'name');

351 // Обновляем кеш новыми данными

1118 $file_crc32 = crc32(file_get_contents($file->getPathname()));
1119
1120 $files_to_load[] = [

1123 'cache_key' => "{$this->session['language']}_{$file_name}_$dir_index",
1124 'lang_id' => $file_name === 'main', // Только для main.php

1290 private function process_lang_array(array $lang): array
1291 {
1292 $result = [
1293 'errors' => [],
1294 'changes' => [],
1295 'result' => $lang, // Инициализируем результат исходным массивом
1296 ];
1297
1298 foreach ($lang as $name => $keys) {
1299 // Проверка ключей первого уровня
1300 if (!is_string($name)) {
1301 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
1302 continue;
1303 }
1304
1305 // Проверка второго уровня
1306 if (!is_array($keys)) {
1307 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
1308 continue;
1309 }
1310
1311 foreach ($keys as $key => $value) {
1312 // Преобразование ключа в строку
1313 $test_key = (string)$key;
1314 if (empty($test_key)) {
1315 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
1316 $key
1317 );
1318 continue;
1319 }
1320
1321 // Удаление старого ключа и добавление нового
1322 unset($result['result'][$name][$key]);
1323 $key = $test_key;
1324
1325 // Проверка значений второго уровня
1326 if (!is_string($value)) {
1327 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
1328 $value
1329 );
1330 continue;
1331 }
1332
1333 // Очистка значения
1334 $original_value = $value;
1335 $cleaned_value = self::clean_field($value);
1336 if ($original_value !== $cleaned_value) {
1337 $result['changes']["{$name}['$key']"] = [
1338 'original' => $original_value,
1339 'cleaned' => $cleaned_value,
1340 ];
1341 $value = $cleaned_value; // Обновляем значение
1342 }
1343
1344 // Добавляем обработанный ключ и значение в результирующий массив
1345 $result['result'][$name][$key] = $value;
1346 }
1347 }
