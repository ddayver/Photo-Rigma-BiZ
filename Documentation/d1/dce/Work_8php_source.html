1038 private function process_lang_array(array $lang): array
1039 {
1040 $result = [
1041 'errors' => [],
1042 'changes' => [],
1043 'result' => $lang, // Инициализируем результат исходным массивом
1044 ];
1045
1046 foreach ($lang as $name => $keys) {
1047 // Проверка ключей первого уровня
1048 if (!is_string($name)) {
1049 $result['errors'][] = "Ключ первого уровня должен быть строкой. Получено: " . gettype($name);
1050 continue;
1051 }
1052
1053 // Проверка второго уровня
1054 if (!is_array($keys)) {
1055 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
1056 continue;
1057 }
1058
1059 foreach ($keys as $key => $value) {
1060 // Преобразование ключа в строку
1061 $test_key = (string)$key;
1062 if (empty($test_key)) {
1063 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
1064 $key
1065 );
1066 continue;
1067 }
1068
1069 // Удаление старого ключа и добавление нового
1070 unset($result['result'][$name][$key]);
1071 $key = $test_key;
1072
1073 // Проверка значений второго уровня
1074 if (!is_string($value)) {
1075 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
1076 $value
1077 );
1078 continue;
1079 }
1080
1081 // Очистка значения
1082 $original_value = $value;
1083 $cleaned_value = self::clean_field($value);
1084 if ($original_value !== $cleaned_value) {
1085 $result['changes']["{$name}['$key']"] = [
1086 'original' => $original_value,
1087 'cleaned' => $cleaned_value,
1088 ];
1089 $value = $cleaned_value; // Обновляем значение
1090 }
1091
1092 // Добавляем обработанный ключ и значение в результирующий массив
1093 $result['result'][$name][$key] = $value;
1094 }
1095 }
1096
1097 return $result;
1098 }
