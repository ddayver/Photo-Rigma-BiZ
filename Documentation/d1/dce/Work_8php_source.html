348 private function _load_cached_config(Database_Interface $db): array
349 {
350 // Получаем время последнего изменения таблицы TBL_CONFIG из таблицы TBL_CHANGE_TIMESTAMP
351 $table_name = str_replace('`', '', TBL_CONFIG); // Убираем символы экранирования
352 $db->select('last_update', TBL_CHANGE_TIMESTAMP, [
353 'where' => '`table_name` = :table_name',
354 'params' => [':table_name' => $table_name],
355 ]);
356 $timestamp_result = $db->result_row();
357
358 // Если время последнего изменения таблицы не найдено, выбрасываем исключение
359 if (!$timestamp_result || !isset($timestamp_result['last_update'])) {
360 throw new RuntimeException(
361 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить дату последнего изменения таблицы: $table_name"
362 );
363 }
364
365 // Преобразуем дату последнего изменения в timestamp для использования в качестве контрольной суммы
366 $db_last_update = strtotime($timestamp_result['last_update']);
367
368 // Проверяем актуальность кеша
369 $config_data = $this->cache->is_valid('config_cache', $db_last_update);
370 if (!$config_data) {
371 // Если кеш устарел или отсутствует, загружаем данные из БД
372 $db->select('*', TBL_CONFIG, ['params' => []]);
373 $result = $db->result_array();
374
375 // Загружаем все данные из таблицы конфигурации
376 if ($result === false) {
377 throw new RuntimeException(
378 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | Не удалось загрузить данные из таблицы: ' . TBL_CONFIG
379 );
380 }
381
382 // Преобразуем данные в ассоциативный массив, где ключи — имена параметров, значения — их значения
383 $config_data = array_column($result, 'value', 'name');
384
385 // Обновляем кеш новыми данными
386 if (!$this->cache->update_cache('config_cache', $db_last_update, $config_data)) {
387 throw new RuntimeException(
388 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | Не удалось записать данные в кеш: config_cache'
389 );
390 }
391 }
392
393 // Получение версии базы данных
394 $db->select('ver', 'db_version', ['limit' => 1]);
395
396 $version_data = $db->result_row();
397
398 if (!is_array($version_data) || !isset($version_data['ver'])) {
399 throw new RuntimeException(
400 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: 'global') . ') | ' . 'Не удалось получить корректную версию из таблицы db_version'
401 );
402 }
403
405 $config_data['db_version'] = $version_data['ver'];
406
407 // Чтение версии приложения из файла ver.php через парсинг содержимого
408 $version_file = $this->config['inc_dir'] . 'ver.php';
409
410 if (!is_file($version_file)) {
411 throw new RuntimeException(
412 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: 'global') . ") | Не найден файл версии: $version_file"
413 );
414 }
415
416 $content = file_get_contents($version_file);
417
418 // Регулярное выражение для поиска строки вида: $ver = '...';
419 $pattern = '/^\\$ver\s*=\s*[\'"]([^\'"]+)[\'"]\s*;?\s*$/m';
420
421 if (!preg_match($pattern, $content, $matches)) {
422 throw new RuntimeException(
423 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: 'global') . ") | Недопустимое содержимое файла версии. Должна быть только строковая переменная \$ver."
424 );
425 }

428 $ver = $matches[1];
429
430 if (!is_string($ver) || trim($ver) === '') {
431 throw new RuntimeException(
432 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: 'global') . ") | Значение \$ver должно быть не пустой строкой"
433 );
434 }

436 $config_data['code_version'] = $ver;

1199 protected function _set_lang_internal(): void
1200 {
1201 // 1. Формирование массива путей
1202 $this->session['language'] = $this->session['language'] ?? $this->config['language'];
1203 $lang_files_path = array_map(
1204 fn (string $dir): string => rtrim($dir, '/') . '/' . $this->session['language'] . '/',
1205 $this->config['language_dirs'] ?? [$this->config['site_dir'] . 'language/']
1206 );
1207
1208 // 2. Сканирование папок и формирование массива файлов для загрузки
1209 $files_to_load = [];
1210 foreach ($lang_files_path as $dir_index => $dir) {
1211 if (!is_dir($dir)) {
1212 log_in_file(
1213 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1214 'Папка не найдена | Путь: ' . $dir
1215 );
1216 continue;
1217 }
1218
1219 foreach (new DirectoryIterator($dir) as $file) {
1220 if (!$file->isFile() || str_ends_with($file->getFilename(), 'index.php')) {
1221 continue; // Пропускаем неподходящие файлы
1222 }
1223
1224 if (!str_ends_with($file->getFilename(), '.php')) {
1225 continue; // Пропускаем файлы без расширения .php
1226 }
1227
1228 $file_name = $file->getBasename('.php');
1229 $file_crc32 = crc32(file_get_contents($file->getPathname()));
1230
1231 $files_to_load[] = [
1232 'path' => $file->getPathname(),
1233 'crc32' => $file_crc32,
1234 'cache_key' => "{$this->session['language']}_{$file_name}_$dir_index",
1235 'lang_id' => $file_name === 'main', // Только для main.php
1236 'throw' => $file_name === 'main' && empty($files_to_load), // Только для первого main.php
1237 ];
1238 }
1239 }
1240
1241 // Проверка: если список файлов пустой, выбрасываем исключение
1242 if (empty($files_to_load)) {
1243 throw new RuntimeException(
1244 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1245 'Не найдено ни одного подходящего файла языковых данных'
1246 );
1247 }
1248
1249 // Проверка: если нет файла main.php, выбрасываем исключение
1250 if (!array_filter($files_to_load, static fn ($file) => $file['throw'])) {
1251 throw new RuntimeException(
1252 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1253 'Не найден обязательный файл main.php'
1254 );
1255 }
1256
1257 // 3. Обработка файлов
1258 $new_lang_data = [];
1259 foreach ($files_to_load as $file) {
1260 // Загрузка данных из файла
1261 $data = include($file['path']);
1262 if (!is_array($data) || !isset($data['lang'])) {
1263 if ($file['throw']) {
1264 throw new RuntimeException(
1265 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1266 'Файл языковых данных не возвращает массив с ключом \'lang\' | Путь: ' . $file['path']
1267 );
1268 }
1269
1270 log_in_file(
1271 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1272 'Некорректные данные в файле | Путь: ' . $file['path']
1273 );
1274 continue;
1275 }
1276
1277 // Обработка lang_id (если требуется)
1278 if ($file['lang_id']) {
1279 $lang_id = $data['lang_id'] ?? null;
1280 if (!is_string($lang_id)) {
1281 if ($file['throw']) {
1282 throw new RuntimeException(
1283 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1284 'Некорректный или отсутствующий lang_id в файле | Путь: ' . $file['path']
1285 );
1286 }
1287
1288 log_in_file(
1289 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1290 'Некорректный или отсутствующий lang_id в файле | Путь: ' . $file['path']
1291 );
1292 continue;
1293 }
1294 // Обновляем значение html_lang
1295 $this->config['html_lang'] = $lang_id;
1296 }
1297
1298 // Проверка кеша
1299 $cached_data = $this->cache->is_valid($file['cache_key'], $file['crc32']);
1300 if ($cached_data !== false) {
1301 // Если кеш актуален, используем данные из кеша
1302 $new_lang_data = [...($new_lang_data ?? []), ...$cached_data];
1303 continue;
1304 }
1305

1307 $result = $this->_process_lang_array($data['lang']);
1308 if (!empty($result['errors'])) {
1309 log_in_file(
1310 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1311 'Обнаружены ошибки в массиве языковых данных | Ошибки: ' .
1312 json_encode($result['errors'], JSON_THROW_ON_ERROR | JSON_UNESCAPED_UNICODE | JSON_PRETTY_PRINT)
1313 );
1314 if ($file['throw']) {
1315 throw new InvalidArgumentException(
1316 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .
1317 'Обнаружены ошибки в массиве языковых данных | Путь: ' . $file['path']
1318 );
1319 }
1320 }
1321
1322 // Логирование изменений
1323 if (!empty($result['changes'])) {
1324 log_in_file(

1327 json_encode($result['changes'], JSON_THROW_ON_ERROR | JSON_UNESCAPED_UNICODE | JSON_PRETTY_PRINT)
1328 );
1329 }
1330

1332 if (!$this->cache->update_cache($file['cache_key'], $file['crc32'], $result['result'])) {
1333 throw new RuntimeException(
1334 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | ' .

1336 );

1340 $new_lang_data = [...($new_lang_data ?? []), ...$result['result']];
1341 }

1402 private function _process_lang_array(array $lang): array
1403 {
1404 $result = [
1405 'errors' => [],
1406 'changes' => [],
1407 'result' => $lang, // Инициализируем результат исходным массивом
1408 ];
1409
1410 foreach ($lang as $name => $keys) {
1411 // Проверка ключей первого уровня
1412 if (!is_string($name)) {
1413 $result['errors'][] = 'Ключ первого уровня должен быть строкой. Получено: ' . gettype($name);
1414 continue;
1415 }
1416
1417 // Проверка второго уровня
1418 if (!is_array($keys)) {
1419 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
1420 continue;
1421 }
1422
1423 foreach ($keys as $key => $value) {
1424 // Преобразование ключа в строку
1425 $test_key = (string)$key;
1426 if (empty($test_key)) {
1427 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
1428 $key
1429 );
1430 continue;
1431 }
1432
1433 // Удаление старого ключа и добавление нового
1434 unset($result['result'][$name][$key]);
1435 $key = $test_key;
1436
1437 // Проверка значений второго уровня
1438 if (!is_string($value)) {
1439 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
1440 $value
1441 );
1442 continue;
1443 }
1444
1445 // Очистка значения
1446 $original_value = $value;
1447 $cleaned_value = self::clean_field($value);
1448 if ($original_value !== $cleaned_value) {
1449 $result['changes']["{$name}['$key']"] = [
1450 'original' => $original_value,
1451 'cleaned' => $cleaned_value,
1452 ];
1453 $value = $cleaned_value; // Обновляем значение
1454 }
1455
1456 // Добавляем обработанный ключ и значение в результирующий массив
1457 $result['result'][$name][$key] = $value;
1458 }
1459 }
1460
1461 return $result;
1462 }
