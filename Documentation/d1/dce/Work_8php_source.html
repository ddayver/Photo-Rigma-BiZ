321 private function load_cached_config(Database_Interface $db): array
322 {
323 // Получаем время последнего изменения таблицы TBL_CONFIG из таблицы TBL_CHANGE_TIMESTAMP
324 $table_name = str_replace('`', '', TBL_CONFIG); // Убираем символы экранирования
325 $db->select('last_update', TBL_CHANGE_TIMESTAMP, [
326 'where' => '`table_name` = :table_name',
327 'params' => [':table_name' => $table_name],
328 ]);
329 $timestamp_result = $db->res_row();
330
331 // Если время последнего изменения таблицы не найдено, выбрасываем исключение
332 if (!$timestamp_result || !isset($timestamp_result['last_update'])) {
333 throw new RuntimeException(
334 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить дату последнего изменения таблицы: $table_name"

336 }
337
338 // Преобразуем дату последнего изменения в timestamp для использования в качестве контрольной суммы
339 $db_last_update = strtotime($timestamp_result['last_update']);
340
341 // Проверяем актуальность кеша
342 $config_data = $this->cache->is_valid('config_cache', $db_last_update);
343 if (!$config_data) {
344 // Если кеш устарел или отсутствует, загружаем данные из БД
345 $db->select('*', TBL_CONFIG, ['params' => []]);

1297 private function process_lang_array(array $lang): array
1298 {
1299 $result = [
1300 'errors' => [],
1301 'changes' => [],
1302 'result' => $lang, // Инициализируем результат исходным массивом
1303 ];
1304
1305 foreach ($lang as $name => $keys) {
1306 // Проверка ключей первого уровня
1307 if (!is_string($name)) {
1308 $result['errors'][] = 'Ключ первого уровня должен быть строкой. Получено: ' . gettype($name);
1309 continue;
1310 }
1311
1312 // Проверка второго уровня
1313 if (!is_array($keys)) {
1314 $result['errors'][] = "Значение для ключа '$name' должно быть массивом.";
1315 continue;
1316 }
1317
1318 foreach ($keys as $key => $value) {
1319 // Преобразование ключа в строку
1320 $test_key = (string)$key;
1321 if (empty($test_key)) {
1322 $result['errors'][] = "Ключ второго уровня для '$name' должен быть строкой. Получено: $key - " . gettype(
1323 $key
1324 );
1325 continue;
1326 }
1327
1328 // Удаление старого ключа и добавление нового
1329 unset($result['result'][$name][$key]);
1330 $key = $test_key;
1331
1332 // Проверка значений второго уровня
1333 if (!is_string($value)) {
1334 $result['errors'][] = "Значение для ключа '{$name}['$key']' должно быть строкой. Получено: " . gettype(
1335 $value
1336 );
1337 continue;
1338 }
1339
1340 // Очистка значения
1341 $original_value = $value;
1342 $cleaned_value = self::clean_field($value);
1343 if ($original_value !== $cleaned_value) {
1344 $result['changes']["{$name}['$key']"] = [
1345 'original' => $original_value,
1346 'cleaned' => $cleaned_value,
1347 ];
1348 $value = $cleaned_value; // Обновляем значение
1349 }
1350
1351 // Добавляем обработанный ключ и значение в результирующий массив
1352 $result['result'][$name][$key] = $value;
