626 protected function _check_input_internal(string $source_name, string $field, array $options = []): bool
627 {
628 $allowed_sources = ['_GET', '_POST', '_SESSION', '_COOKIE', '_FILES'];
629 if (!in_array($source_name, $allowed_sources, true)) {
630 log_in_file(
631 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый источник данных для check_input | Получено: $source_name"
632 );
633 return false;
634 }
635 $source = match ($source_name) {
636 '_GET' => $_GET,
637 '_POST' => $_POST,
638 '_SESSION' => $this->session,
639 '_COOKIE' => $_COOKIE,
640 '_FILES' => $_FILES,
641 default => null,
642 };
643 if (!is_array($source)) {
644 log_in_file(
645 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Источник данных не является массивом | Источник: $source_name"
646 );
647 return false;
648 }
649 if (($options['isset'] ?? false) && !isset($source[$field])) {
650 return false;
651 }
652 if (($options['empty'] ?? false) && empty($source[$field])) {
653 return false;
654 }
655 if ($source_name === '_FILES') {
656 if (!isset($source[$field])) {
657 return false;
658 }
659 $file = $source[$field];
660 if ($file['error'] !== UPLOAD_ERR_OK) {
661 log_in_file(
662 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка загрузки файла для поля | Поле: $field"
663 );
664 return false;
665 }
666 $max_size = $options['max_size'] ?? 0;
667 if ($max_size > 0 && $file['size'] > $max_size) {
668 log_in_file(
669 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размер файла превышает максимально допустимый размер | Поле: $field, Размер: {$file['size']}, Максимальный размер: $max_size"
670 );
671 return false;
672 }
673 $finfo = finfo_open(FILEINFO_MIME_TYPE);
674 $real_mime_type = finfo_file($finfo, $file['tmp_name']);
675 finfo_close($finfo);
676 if (!Work::validate_mime_type($real_mime_type)) {
677 throw new RuntimeException(
678 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый MIME-тип для загруженного файла | Поле: $field, MIME-тип: $real_mime_type"
679 );
680 }
681 return true;
682 }
683 return $this->_check_field_internal(
684 $source[$field] ?? null,
685 $options['regexp'] ?? false,
686 $options['not_zero'] ?? false
687 );
688 }
