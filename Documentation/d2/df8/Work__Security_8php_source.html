635 protected function _check_input_internal(string $source_name, string $field, array $options = []): bool
636 {
637 $allowed_sources = ['_GET', '_POST', '_SESSION', '_COOKIE', '_FILES'];
638 if (!in_array($source_name, $allowed_sources, true)) {
639 log_in_file(
640 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый источник данных для check_input | Получено: $source_name"
641 );
642 return false;
643 }
644 $source = match ($source_name) {
645 '_GET' => $_GET,
646 '_POST' => $_POST,
647 '_SESSION' => $this->session,
648 '_COOKIE' => $_COOKIE,
649 '_FILES' => $_FILES,
650 default => null,
651 };
652 if (!is_array($source)) {
653 log_in_file(
654 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Источник данных не является массивом | Источник: $source_name"
655 );
656 return false;
657 }
658 if (($options['isset'] ?? false) && !isset($source[$field])) {
659 return false;
660 }
661 if (($options['empty'] ?? false) && empty($source[$field])) {
662 return false;
663 }
664 if ($source_name === '_FILES') {
665 if (!isset($source[$field])) {
666 return false;
667 }
668 $file = $source[$field];
669 if ($file['error'] !== UPLOAD_ERR_OK) {
670 log_in_file(
671 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка загрузки файла для поля | Поле: $field"
672 );
673 return false;
674 }
675 $max_size = $options['max_size'] ?? 0;
676 if ($max_size > 0 && $file['size'] > $max_size) {
677 log_in_file(
678 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размер файла превышает максимально допустимый размер | Поле: $field, Размер: {$file['size']}, Максимальный размер: $max_size"
679 );
680 return false;
681 }
682 $finfo = finfo_open(FILEINFO_MIME_TYPE);
683 $real_mime_type = finfo_file($finfo, $file['tmp_name']);
684 finfo_close($finfo);
685 if (!Work::validate_mime_type($real_mime_type)) {
686 throw new RuntimeException(
687 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый MIME-тип для загруженного файла | Поле: $field, MIME-тип: $real_mime_type"
688 );
689 }
690 return true;
691 }
692 return $this->_check_field_internal(
693 $source[$field] ?? null,
694 $options['regexp'] ?? false,
695 $options['not_zero'] ?? false
696 );
697 }
