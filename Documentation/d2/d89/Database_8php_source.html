793 private function build_conditions(array $options): array
794 {
795 $conditions = '';
796 $params = $options['params'] ?? [];
797
798 // === 1. Обработка WHERE ===
799 if (isset($options['where'])) {
800 if (is_string($options['where'])) {
801 // Если where — строка, добавляем её напрямую
802 $conditions .= ' WHERE ' . $options['where'];
803 } elseif (is_array($options['where'])) {
804 // Если where — массив, обрабатываем его
805 $conditions .= ' WHERE ' . implode(' AND ', $options['where']);
806 } else {
807 throw new InvalidArgumentException(
808 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное условие 'where' | Ожидалась строка или массив, получено: " . gettype(
809 $options['where']
810 )
811 );
812 }
813 }
814
815 // === 2. Обработка GROUP BY ===
816 if (isset($options['group'])) {
817 if (is_string($options['group'])) {
818 $conditions .= ' GROUP BY ' . $options['group'];
819 } else {
820 throw new InvalidArgumentException(
821 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное значение 'group' | Ожидалась строка, получено: " . gettype(
822 $options['group']
823 )
824 );
825 }
826 }
827
828 // === 3. Обработка ORDER BY ===
829 if (isset($options['order'])) {
830 if (is_string($options['order'])) {
831 $conditions .= ' ORDER BY ' . $options['order'];
832 } else {
833 throw new InvalidArgumentException(
834 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное значение 'order' | Ожидалась строка, получено: " . gettype(
835 $options['order']
836 )
837 );
838 }
839 }
840
841 // === 4. Обработка LIMIT ===
842 if (isset($options['limit'])) {
843 if (is_numeric($options['limit'])) {
844 $conditions .= ' LIMIT ' . (int)$options['limit'];
845 } elseif (is_string($options['limit']) && preg_match('/^\d+\s*,\s*\d+$/', $options['limit'])) {
846 [$offset, $count] = array_map('intval', explode(',', $options['limit']));
847 $conditions .= ' LIMIT ' . $offset . ', ' . $count;
848 } else {
849 throw new InvalidArgumentException(
850 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Неверное значение 'limit' | Ожидалось число или строка формата 'OFFSET, COUNT', получено: " . gettype(
851 $options['limit']
852 ) . " ($options[limit])"
853 );
854 }
855 }
856
857 return [$conditions, $params];

963 private function execute_query(array $params = []): bool
964 {
965 // Валидация состояния запроса и аргументов
966 if (isset($this->res_query) && !($this->res_query instanceof PDOStatement)) {
967 throw new InvalidArgumentException(
968 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректное состояние результата запроса | Текущее значение: " . gettype(
969 $this->res_query
970 )
971 );
972 }
973 // Очистка внутренних свойств для вывода результата запроса
974 $this->res_query = null;
975 $this->aff_rows = 0;
976 $this->insert_id = 0;
977 // Валидация аргументов
978 if (!is_string($this->txt_query)) {
979 throw new InvalidArgumentException(
980 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверный тип SQL-запроса | Ожидалась строка, получено: " . gettype(
981 $this->txt_query
982 )
983 );
984 }
985
986 // Финальное преобразование запроса в нужный формат СУБД.
987 $this->convert_query();
988
989 // Выполняем EXPLAIN ANALYZE, если включено
990 if (SQL_ANALYZE) {
991 $this->log_explain($params);
992 }
993
994 // Начало замера времени
995 $start_time = microtime(true);
996 $this->res_query = $this->pdo->prepare($this->txt_query);
997 $this->res_query->execute($params);
998 $this->aff_rows = $this->res_query->rowCount();
999 // Проверяем, является ли запрос INSERT
1000 if (stripos(trim($this->txt_query), 'INSERT') === 0) {
1001 $this->insert_id = (int)$this->pdo->lastInsertId();
1002 }
1003 // Конец замера времени
1004 $end_time = microtime(true);
1005 $execution_time = ($end_time - $start_time) * 1000; // Время в миллисекундах
1006 // Логирование медленных запросов
1007 $slow_query_threshold = 200; // Порог в миллисекундах
1008 if (SQL_LOG && $execution_time > $slow_query_threshold) {
1009 // Сохраняем запрос в БД
1010 $this->log_query('slow', $execution_time);
1011 } elseif (SQL_LOG && empty($params)) {
1012 // Сохраняем запрос в БД с пустым плейсхолдером
1013 $this->log_query('no_placeholders', $execution_time);
1014 }
1015 // Очистка строки запроса после её выполнения
1016 $this->txt_query = null;
1017 return true;
1018 }

1148 private function log_explain(array $params): void
1149 {
1150 $explain = '';
1151 try {
1152 // Определяем, нужно ли добавлять ANALYZE
1153 $is_write_query = stripos(trim($this->txt_query), 'INSERT') === 0 || stripos(
1154 trim($this->txt_query),
1155 'UPDATE'
1156 ) === 0 || stripos(trim($this->txt_query), 'DELETE') === 0 || stripos(
1157 trim($this->txt_query),
1158 'TRUNCATE'
1159 ) === 0;
1160
1161 // Определяем тип EXPLAIN в зависимости от СУБД и версии
1162 if ($this->db_type === 'sqlite') {
1163 $explain = 'EXPLAIN'; // Для SQLite всегда только EXPLAIN
1164 } elseif ($this->db_type === 'mysql') {
1165 // Для MySQL или MariaDB проверяем версию и тип СУБД
1166 try {
1167 $stmt = $this->pdo->query("SELECT VERSION()");
1168 $version = $stmt->fetchColumn();
1169
1170 // Проверяем, является ли СУБД MariaDB
1171 $is_mariadb = stripos($version, 'mariadb') !== false;
1172
1173 // Извлекаем числовую часть версии (например, "8.0.30" -> 8.0)
1174 preg_match('/^(\d+\.\d+)/', $version, $matches);
1175 $numeric_version = $matches[1] ?? null;
1176
1177 // Определяем, поддерживается ли ANALYZE
1178 $supports_analyze = !$is_mariadb && $numeric_version !== null && version_compare(
1179 $numeric_version,
1180 '5.7',
1181 '>='
1182 );
1183
1184 $explain = $is_write_query || !$supports_analyze ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE или для MariaDB/MySQL < 5.7
1185 : 'EXPLAIN ANALYZE';
1186 } catch (Exception) {
1187 // Если не удалось получить версию, используем только EXPLAIN
1188 $explain = 'EXPLAIN';
1189 }
1190 } elseif ($this->db_type === 'pgsql') {
1191 $explain = $is_write_query ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE
1192 : 'EXPLAIN ANALYZE'; // С ANALYZE для SELECT
1193 } else {
1194 throw new InvalidArgumentException(
1195 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не поддерживаемый тип СУБД | Тип: $this->db_type"
1196 );
1197 }
1198
1199 // Выполняем EXPLAIN (или EXPLAIN ANALYZE)
1200 $stmt = $this->pdo->prepare("$explain $this->txt_query");
1201 $stmt->execute($params);
1202
1203 // Получаем результат
1204 $explain_rows = $stmt->fetchAll(PDO::FETCH_ASSOC);
1205
1206 // Форматируем результат для логирования
1207 $explain_result = '';
1208 foreach ($explain_rows as $row) {
1209 $explain_result .= implode(
1210 ' | ',
1211 array_map(static fn ($key, $value) => "$key: $value", array_keys($row), $row)
1212 ) . PHP_EOL;
1213 }
1214
1215 // Логируем результат
1216 log_in_file(
1217 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Отчет по анализу с помощью $explain: | " . "Запрос: " . $this->txt_query . PHP_EOL . "Параметры: " . (!empty($params) ? json_encode(
1218 $params,
1219 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1220 ) : "отсутствуют") . PHP_EOL . "$explain Результат:" . PHP_EOL . $explain_result
1221 );
1222 } catch (Exception $e) {
1223 // Логируем ошибку
1224 log_in_file(
1225 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при выполнении $explain: | " . "Запрос: " . $this->txt_query . PHP_EOL . "Параметры: " . (!empty($params) ? json_encode(
1226 $params,
1227 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1228 ) : "отсутствуют") . PHP_EOL . "Сообщение об ошибке: {$e->getMessage()}"
1229 );
1230 }

1301 private function log_query(string $reason = 'other', ?float $execution_time_ms = null): void
1302 {
1303 // Проверяем, что запрос существует, является строкой и не является запросом к таблице логов
1304 if (!is_string($this->txt_query) || empty($this->txt_query) || str_contains($this->txt_query, TBL_LOG_QUERY)) {
1305 return;
1306 }
1307
1308 // Проверяем и нормализуем время выполнения (минимальное значение 0)
1309 $execution_time_ms = max(0.0, (float)($execution_time_ms ?? 0));
1310
1311 // Обрезаем запрос до допустимой длины (например, 65,530 символа для TEXT в MySQL)
1312 $max_query_length = 65530; // Максимальная длина для TEXT
1313 $log_txt_query = strlen($this->txt_query) > $max_query_length ? substr(
1314 $this->txt_query,
1315 0,
1316 $max_query_length
1317 ) . '...' // Добавляем суффикс, если текст обрезан
1318 : $this->txt_query;
1319
1320 // Проверяем значение $reason через match
