719 private function build_conditions(array $options): array
720 {
721 $conditions = '';
722 $params = $options['params'] ?? [];
723
724 // === 1. Обработка WHERE ===
725 if (isset($options['where'])) {
726 if (is_string($options['where'])) {
727 // Если where — строка, добавляем её напрямую
728 $conditions .= ' WHERE ' . $options['where'];
729 } elseif (is_array($options['where'])) {

734 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное условие 'where' | Ожидалась строка или массив, получено: " . gettype(
735 $options['where']
736 )
737 );
738 }
739 }
740
741 // === 2. Обработка GROUP BY ===
742 if (isset($options['group'])) {
743 if (is_string($options['group'])) {
744 $conditions .= ' GROUP BY ' . $options['group'];
745 } else {
746 throw new InvalidArgumentException(
747 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное значение 'group' | Ожидалась строка, получено: " . gettype(
748 $options['group']
749 )
750 );
751 }
752 }
753
754 // === 3. Обработка ORDER BY ===
755 if (isset($options['order'])) {
756 if (is_string($options['order'])) {
757 $conditions .= ' ORDER BY ' . $options['order'];
758 } else {
759 throw new InvalidArgumentException(
760 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверное значение 'order' | Ожидалась строка, получено: " . gettype(
761 $options['order']
762 )
763 );
764 }
765 }
766
767 // === 4. Обработка LIMIT ===
768 if (isset($options['limit'])) {
769 if (is_numeric($options['limit'])) {
770 $conditions .= ' LIMIT ' . (int)$options['limit'];
771 } elseif (is_string($options['limit']) && preg_match('/^\d+\s*,\s*\d+$/', $options['limit'])) {
772 [$offset, $count] = array_map('intval', explode(',', $options['limit']));
773 $conditions .= ' LIMIT ' . $offset . ', ' . $count;
774 } else {
775 throw new InvalidArgumentException(
776 "[{__FILE__}:{__LINE__} ({__METHOD__ ?: __FUNCTION__ ?: 'global'})] | Неверное значение 'limit' | Ожидалось число или строка формата 'OFFSET, COUNT', получено: " . gettype(
777 $options['limit']
778 ) . " ($options[limit])"
779 );
780 }
781 }

889 private function execute_query(array $params = []): bool
890 {
891 // Валидация состояния запроса и аргументов
892 if (isset($this->res_query) && !($this->res_query instanceof PDOStatement)) {
893 throw new InvalidArgumentException(
894 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректное состояние результата запроса | Текущее значение: " . gettype(
895 $this->res_query
896 )
897 );
898 }
899 // Очистка внутренних свойств для вывода результата запроса
900 $this->res_query = null;
901 $this->aff_rows = 0;
902 $this->insert_id = 0;
903 // Валидация аргументов
904 if (!is_string($this->txt_query)) {
905 throw new InvalidArgumentException(
906 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Неверный тип SQL-запроса | Ожидалась строка, получено: " . gettype(
907 $this->txt_query
908 )
909 );
910 }
911
912 // Финальное преобразование запроса в нужный формат СУБД.
913 $this->convert_query();
914
915 // Выполняем EXPLAIN ANALYZE, если включено
916 if (SQL_ANALYZE) {
917 $this->log_explain($params);
918 }
919
920 // Начало замера времени
921 $start_time = microtime(true);
922 $this->res_query = $this->pdo->prepare($this->txt_query);
923 $this->res_query->execute($params);
924 $this->aff_rows = $this->res_query->rowCount();
925 // Проверяем, является ли запрос INSERT
926 if (stripos(trim($this->txt_query), 'INSERT') === 0) {
927 $this->insert_id = (int)$this->pdo->lastInsertId();
928 }
929 // Конец замера времени
930 $end_time = microtime(true);
931 $execution_time = ($end_time - $start_time) * 1000; // Время в миллисекундах
932 // Логирование медленных запросов
933 $slow_query_threshold = 200; // Порог в миллисекундах
934 if (SQL_LOG && $execution_time > $slow_query_threshold) {
935 // Сохраняем запрос в БД
936 $this->log_query('slow', $execution_time);
937 } elseif (SQL_LOG && empty($params)) {
938 // Сохраняем запрос в БД с пустым плейсхолдером
939 $this->log_query('no_placeholders', $execution_time);
940 }
941 // Очистка строки запроса после её выполнения
942 $this->txt_query = null;
943 return true;
944 }

1065 private function log_explain(array $params): void
1066 {
1067 $explain = '';
1068 try {
1069 // Определяем, нужно ли добавлять ANALYZE
1070 $is_write_query = stripos(trim($this->txt_query), 'INSERT') === 0 || stripos(
1071 trim($this->txt_query),
1072 'UPDATE'
1073 ) === 0 || stripos(trim($this->txt_query), 'DELETE') === 0 || stripos(
1074 trim($this->txt_query),
1075 'TRUNCATE'
1076 ) === 0;
1077
1078 // Определяем тип EXPLAIN в зависимости от СУБД и версии
1079 if ($this->db_type === 'sqlite') {
1080 $explain = 'EXPLAIN'; // Для SQLite всегда только EXPLAIN
1081 } elseif ($this->db_type === 'mysql') {
1082 // Для MySQL или MariaDB проверяем версию и тип СУБД
1083 try {
1084 $stmt = $this->pdo->query("SELECT VERSION()");
1085 $version = $stmt->fetchColumn();
1086
1087 // Проверяем, является ли СУБД MariaDB
1088 $is_mariadb = stripos($version, 'mariadb') !== false;
1089
1090 // Извлекаем числовую часть версии (например, "8.0.30" -> 8.0)
1091 preg_match('/^(\d+\.\d+)/', $version, $matches);
1092 $numeric_version = $matches[1] ?? null;
1093
1094 // Определяем, поддерживается ли ANALYZE
1095 $supports_analyze = !$is_mariadb && $numeric_version !== null && version_compare(
1096 $numeric_version,
1097 '5.7',
1098 '>='
1099 );
1100
1101 $explain = $is_write_query || !$supports_analyze ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE или для MariaDB/MySQL < 5.7
1102 : 'EXPLAIN ANALYZE';
1103 } catch (Exception) {
1104 // Если не удалось получить версию, используем только EXPLAIN
1105 $explain = 'EXPLAIN';
1106 }
1107 } elseif ($this->db_type === 'pgsql') {
1108 $explain = $is_write_query ? 'EXPLAIN' // Без ANALYZE для INSERT/UPDATE/DELETE/TRUNCATE
1109 : 'EXPLAIN ANALYZE'; // С ANALYZE для SELECT
1110 } else {
1111 throw new InvalidArgumentException(
1112 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не поддерживаемый тип СУБД | Тип: $this->db_type"
1113 );
1114 }
1115
1116 // Выполняем EXPLAIN (или EXPLAIN ANALYZE)
1117 $stmt = $this->pdo->prepare("$explain $this->txt_query");
1118 $stmt->execute($params);
1119
1120 // Получаем результат
1121 $explain_rows = $stmt->fetchAll(PDO::FETCH_ASSOC);
1122
1123 // Форматируем результат для логирования
1124 $explain_result = '';
1125 foreach ($explain_rows as $row) {
1126 $explain_result .= implode(
1127 ' | ',
1128 array_map(static fn ($key, $value) => "$key: $value", array_keys($row), $row)
1129 ) . PHP_EOL;
1130 }
1131
1132 // Логируем результат
1133 log_in_file(
1134 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Отчет по анализу с помощью $explain: | " . "Запрос: " . $this->txt_query . PHP_EOL . "Параметры: " . (!empty($params) ? json_encode(
1135 $params,
1136 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1137 ) : "отсутствуют") . PHP_EOL . "$explain Результат:" . PHP_EOL . $explain_result
1138 );
1139 } catch (Exception $e) {
1140 // Логируем ошибку
1141 log_in_file(
1142 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка при выполнении $explain: | " . "Запрос: " . $this->txt_query . PHP_EOL . "Параметры: " . (!empty($params) ? json_encode(
1143 $params,
1144 JSON_THROW_ON_ERROR | JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE
1145 ) : "отсутствуют") . PHP_EOL . "Сообщение об ошибке: {$e->getMessage()}"
1146 );
1147 }

1218 private function log_query(string $reason = 'other', ?float $execution_time_ms = null): void
1219 {
1220 // Проверяем, что запрос существует, является строкой и не является запросом к таблице логов
1221 if (!is_string($this->txt_query) || empty($this->txt_query) || str_contains($this->txt_query, TBL_LOG_QUERY)) {
1222 return;
1223 }
1224
1225 // Проверяем и нормализуем время выполнения (минимальное значение 0)
1226 $execution_time_ms = max(0.0, (float)($execution_time_ms ?? 0));
1227
1228 // Обрезаем запрос до допустимой длины (например, 65,530 символа для TEXT в MySQL)
1229 $max_query_length = 65530; // Максимальная длина для TEXT
1230 $log_txt_query = strlen($this->txt_query) > $max_query_length ? substr(
1231 $this->txt_query,
1232 0,
1233 $max_query_length
1234 ) . '...' // Добавляем суффикс, если текст обрезан
1235 : $this->txt_query;
1236
1237 // Проверяем значение $reason через match
1238 $reason = match ($reason) {
1239 'slow', 'no_placeholders' => $reason, // Оставляем только допустимые значения
1240 default => 'other',
1241 };
1242
1243 // Хэшируем запрос для проверки на дублирование
1244 $hash = md5($this->txt_query);
1245
1246 // Проверяем кэш на наличие запроса
1247 if (!isset($this->query_cache[$hash])) {
1248 // Если запроса нет в кэше, проверяем его наличие в базе данных
1249 $stmt = $this->pdo->prepare(
1250 "SELECT id, usage_count, execution_time FROM " . TBL_LOG_QUERY . " WHERE query_hash = :hash"
1251 );
1252 $stmt->execute([':hash' => $hash]);
1253 $this->query_cache[$hash] = $stmt->fetch(PDO::FETCH_ASSOC); // Сохраняем результат в кэше
1254 }
1255
1256 if ($this->query_cache[$hash]) {
1257 // Если запрос уже существует, обновляем счетчик использования, время последнего использования и максимальное время выполнения
1258 $new_usage_count = $this->query_cache[$hash]['usage_count'] + 1;
1259 $max_execution_time = max(
1260 (float)($this->query_cache[$hash]['execution_time'] ?? 0),
1261 (float)$execution_time_ms
1262 );
1263
1264 $stmt = $this->pdo->prepare(
1265 "UPDATE " . TBL_LOG_QUERY . "
1266 SET usage_count = :usage_count, last_used_at = CURRENT_TIMESTAMP, execution_time = :execution_time
1267 WHERE id = :id"
1268 );
1269 $stmt->execute([
1270 ':usage_count' => $new_usage_count,
1271 ':execution_time' => $max_execution_time,
1272 ':id' => $this->query_cache[$hash]['id'],
1273 ]);
1274 } else {
1275 // Если запрос не существует, сохраняем его в таблицу логов
1276 $stmt = $this->pdo->prepare(
1277 "INSERT INTO " . TBL_LOG_QUERY . " (query_hash, query_text, reason, execution_time)
1278 VALUES (:hash, :text, :reason, :execution_time)"
1279 );
1280 $stmt->execute([
1281 ':hash' => $hash,
1282 ':text' => $log_txt_query,
1283 ':reason' => $reason,
1284 ':execution_time' => $execution_time_ms, // Время в миллисекундах
1285 ]);
1286 }
