485 protected static function _return_bytes_internal(string|int $val): int
486 {
487 // Проверяем, что входные данные являются строкой или числом
488 if (!is_string($val) && !is_numeric($val)) {
489 return 0; // Возвращаем 0 для некорректных данных
490 }
491 // Удаляем лишние пробелы
492 $val = trim((string)$val);
493 // Если строка содержит только цифры, возвращаем значение как есть
494 if (ctype_digit($val)) {
495 return (int)$val;
496 }
497 // Извлекаем последний символ (суффикс)
498 $last = strtolower($val[-1]);
499 // Извлекаем числовую часть (все символы, кроме последнего)
500 $number = substr($val, 0, -1);
501 // Проверяем, является ли числовая часть корректной
502 if (!is_numeric($number)) {
503 return 0; // Возвращаем 0 для некорректных данных
504 }
505 // Преобразуем в абсолютное значение
506 $number = abs((float)$number);
507 // Преобразуем в байты с использованием match
508 return match ($last) {
509 'g' => (int)($number * 1024 * 1024 * 1024),
510 'm' => (int)($number * 1024 * 1024),
511 'k' => (int)($number * 1024),
512 default => (int)$number, // Если суффикс недопустим, возвращаем только числовую часть
513 };

1201 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
1202 {
1203 // Проверка граничных условий
1204 if ($width <= 0 || empty($break)) {
1205 log_in_file(
1206 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = $width, break = '$break'"
1207 );
1208 return $str;
1209 }
1210 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
1211 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
1212 return $str;
1213 }
1214 // Разбиваем строку с использованием wordwrap()
1215 $wrapped = wordwrap($str, $width, $break, true);
1216 // Если строка содержит многобайтовые символы, корректируем разбиение
1217 $lines = explode($break, $wrapped);
1218 $result = [];
1219 foreach ($lines as $line) {
1220 if (mb_strlen($line, 'UTF-8') > $width) {
1221 $result[] = mb_substr($line, 0, $width, 'UTF-8');
1222 $result[] = mb_substr($line, $width, null, 'UTF-8');
1223 } else {
1224 $result[] = $line;
1225 }
1226 }
1227 return implode($break, $result);
1228 }
