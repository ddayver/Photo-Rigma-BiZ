686 protected static function _clean_field_internal($field): ?string
687 {
688 // Если входные данные пусты, возвращаем null
689 if ($field === null || $field === '') {
690 return null;
691 }
692 // Преобразуем данные в строку, если они не являются строкой

751 protected static function _return_bytes_internal($val): int
752 {
753 // Проверяем, что входные данные являются строкой или числом
754 if (!is_string($val) && !is_numeric($val)) {
755 return 0; // Возвращаем 0 для некорректных данных
756 }
757 // Удаляем лишние пробелы
758 $val = trim((string)$val);
759 // Если строка содержит только цифры, возвращаем значение как есть
760 if (ctype_digit($val)) {
761 return (int)$val;
762 }
763 // Извлекаем последний символ (суффикс)
764 $last = strtolower($val[-1]);
765 $number = rtrim($val, 'kmg');
766 // Проверяем, является ли числовая часть корректной
767 if (!is_numeric($number)) {
768 return 0; // Возвращаем 0 для некорректных данных
769 }
770 // Преобразуем в абсолютное значение
771 $number = abs((float)$number);
772 // Преобразуем в байты с использованием match
773 return match ($last) {
774 'g' => (int)($number * 1024 * 1024 * 1024),
775 'm' => (int)($number * 1024 * 1024),
776 'k' => (int)($number * 1024),
777 default => (int)$number, // Если суффикс недопустим, возвращаем только числовую часть

1069 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
1070 {
1071 // Проверка граничных условий
1072 if ($width <= 0 || empty($break)) {
1073 \PhotoRigma\Include\log_in_file(
1074 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = {$width}, break = '{$break}'"
1075 );
1076 return $str;
1077 }
1078 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
1079 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
1080 return $str;
1081 }
1082 // Разбиваем строку с использованием wordwrap()
1083 $wrapped = wordwrap($str, $width, $break, true);
1084 // Если строка содержит многобайтовые символы, корректируем разбиение
1085 $lines = explode($break, $wrapped);
1086 $result = [];
1087 foreach ($lines as $line) {
1088 if (mb_strlen($line, 'UTF-8') > $width) {
1089 $result[] = mb_substr($line, 0, $width, 'UTF-8');
1090 $result[] = mb_substr($line, $width, null, 'UTF-8');
1091 } else {
1092 $result[] = $line;
1093 }
1094 }
1095 return implode($break, $result);
1096 }
