709 protected static function _return_bytes_internal(string|int $val): int
710 {
711 // Проверяем, что входные данные являются строкой или числом
712 if (!is_string($val) && !is_numeric($val)) {
713 return 0; // Возвращаем 0 для некорректных данных
714 }
715 // Удаляем лишние пробелы
716 $val = trim((string)$val);
717 // Если строка содержит только цифры, возвращаем значение как есть
718 if (ctype_digit($val)) {
719 return (int)$val;
720 }
721 // Извлекаем последний символ (суффикс)
722 $last = strtolower($val[-1]);
723 // Извлекаем числовую часть (все символы, кроме последнего)
724 $number = substr($val, 0, -1);
725 // Проверяем, является ли числовая часть корректной
726 if (!is_numeric($number)) {
727 return 0; // Возвращаем 0 для некорректных данных
728 }
729 // Преобразуем в абсолютное значение
730 $number = abs((float)$number);
731 // Преобразуем в байты с использованием match
732 return match ($last) {
733 'g' => (int)($number * 1024 * 1024 * 1024),
734 'm' => (int)($number * 1024 * 1024),
735 'k' => (int)($number * 1024),
736 default => (int)$number, // Если суффикс недопустим, возвращаем только числовую часть

1214 */
1215 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
1216 {
1217 // Проверка граничных условий
1218 if ($width <= 0 || empty($break)) {
1219 log_in_file(
1220 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = $width, break = '$break'"
1221 );
1222 return $str;
1223 }
1224 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
1225 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
1226 return $str;
1227 }
1228 // Разбиваем строку с использованием wordwrap()
1229 $wrapped = wordwrap($str, $width, $break, true);
1230 // Если строка содержит многобайтовые символы, корректируем разбиение
1231 $lines = explode($break, $wrapped);
1232 $result = [];
1233 foreach ($lines as $line) {
1234 if (mb_strlen($line, 'UTF-8') > $width) {
1235 $result[] = mb_substr($line, 0, $width, 'UTF-8');
1236 $result[] = mb_substr($line, $width, null, 'UTF-8');
1237 } else {
1238 $result[] = $line;
1239 }
1240 }
1241 return implode($break, $result);
1242 }
