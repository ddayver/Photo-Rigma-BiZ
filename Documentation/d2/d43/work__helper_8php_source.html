509 'm' => (int)($number * 1024 * 1024),

1124 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
1125 {
1126 // Проверка граничных условий
1127 if ($width <= 0 || empty($break)) {
1128 log_in_file(
1129 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = {$width}, break = '{$break}'"
1130 );
1131 return $str;
1132 }
1133 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
1134 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
1135 return $str;
1136 }
1137 // Разбиваем строку с использованием wordwrap()
1138 $wrapped = wordwrap($str, $width, $break, true);
1139 // Если строка содержит многобайтовые символы, корректируем разбиение
1140 $lines = explode($break, $wrapped);
1141 $result = [];
1142 foreach ($lines as $line) {
1143 if (mb_strlen($line, 'UTF-8') > $width) {
1144 $result[] = mb_substr($line, 0, $width, 'UTF-8');
1145 $result[] = mb_substr($line, $width, null, 'UTF-8');
1146 } else {
1147 $result[] = $line;
1148 }
1149 }
1150 return implode($break, $result);
1151 }
