482 protected static function _clean_field_internal($field): ?string
483 {
484 // Если входные данные пусты, возвращаем null
485 if ($field === null || $field === '') {
486 return null;
487 }
488 // Преобразуем данные в строку, если они не являются строкой
489 if (!is_string($field)) {
490 $field = (string)$field;
491 }
492 // Гарантируем корректную кодировку UTF-8
493 $field = mb_convert_encoding($field, 'UTF-8', 'UTF-8');
494 // Удаляем HTML-теги
495 $field = strip_tags($field);
496 // Экранируем специальные символы с использованием современных флагов
497 $field = htmlspecialchars($field, ENT_QUOTES | ENT_HTML5, 'UTF-8');
498 return $field;
499 }

518 protected static function _return_bytes_internal($val): int
519 {
520 // Проверяем, что входные данные являются строкой или числом
521 if (!is_string($val) && !is_numeric($val)) {
522 return 0; // Возвращаем 0 для некорректных данных
523 }
524 // Удаляем лишние пробелы
525 $val = trim((string)$val);
526 // Если строка содержит только цифры, возвращаем значение как есть
527 if (ctype_digit($val)) {
528 return (int)$val;
529 }
530 // Извлекаем последний символ (суффикс)
531 $last = strtolower($val[-1]);
532 $number = rtrim($val, 'kmg');
533 // Проверяем, является ли числовая часть корректной
534 if (!is_numeric($number)) {
535 return 0; // Возвращаем 0 для некорректных данных
536 }
537 // Преобразуем в абсолютное значение
538 $number = abs((float)$number);
539 // Преобразуем в байты с использованием match
540 return match ($last) {
541 'g' => (int)($number * 1024 * 1024 * 1024),
542 'm' => (int)($number * 1024 * 1024),
543 'k' => (int)($number * 1024),
544 default => (int)$number, // Если суффикс недопустим, возвращаем только числовую часть
545 };
546 }

639 protected static function _ubb_internal(string $text): string
640 {
641 // Очищаем текст от потенциально опасных символов
642 $text = self::clean_field($text);
643 // Максимальная глубина рекурсии для вложенных BBCode
644 $max_recursion_depth = 10;
645 // Рекурсивная функция для обработки BBCode
646 $process_recursively = function ($text, $depth) use (&$process_recursively, $max_recursion_depth) {
647 if ($depth > $max_recursion_depth) {
648 log_in_file(
649 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Превышена максимальная глубина рекурсии при обработке BBCode"
650 );
651 return $text; // Прерываем обработку
652 }
653 // Проверяем, есть ли BBCode-теги
654 if (!preg_match('/\[[a-z].*?\]/', $text)) {
655 return $text; // Нет BBCode-тегов — завершаем обработку
656 }
657 // Паттерны для преобразования BBCode в HTML
658 $patterns = [
659 // Жирный текст
660 '#\[b\](.*?)\[/b\]#si' => fn ($matches) => '<strong>' . $process_recursively($matches[1], $depth + 1) . '</strong>',
661 // Подчёркнутый текст
662 '#\[u\](.*?)\[/u\]#si' => fn ($matches) => '<u>' . $process_recursively($matches[1], $depth + 1) . '</u>',
663 // Курсив
664 '#\[i\](.*?)\[/i\]#si' => fn ($matches) => '<em>' . $process_recursively($matches[1], $depth + 1) . '</em>',
665 // Простая ссылка
666 '#\[url\](.*?)\[/url\]#si' => function ($matches) {
667 $url = self::clean_field($matches[1]);
668 if (
669 !filter_var($url, FILTER_VALIDATE_URL) ||
670 preg_match('/^(javascript|data|vbscript):/i', $url) ||
671 strlen($url) > 2000
672 ) {
673 log_in_file(
674 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный URL | Получено: '$url'"
675 );
676 return '<a href="#" title="#">A-a-a-a!</a>'; // Безопасное значение
677 }
678 return '<a href="' . $url . '" target="_blank" rel="noopener noreferrer" title="' . $url . '">' . $url . '</a>';
679 },
680 // Ссылка с текстом
681 '#\[url=(.*?)\](.*?)\[/url\]#si' => function ($matches) {
682 $url = self::clean_field($matches[1]);
683 $text = self::clean_field($matches[2]);
684 if (
685 !filter_var($url, FILTER_VALIDATE_URL) ||
686 preg_match('/^(javascript|data|vbscript):/i', $url) ||
687 strlen($url) > 2000
688 ) {
689 log_in_file(
690 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный URL | Получено: '$url'"
691 );
692 return '<a href="#" title="#">A-a-a-a!</a>'; // Безопасное значение
693 }
694 return '<a href="' . $url . '" target="_blank" rel="noopener noreferrer" title="' . $text . '">' . $text . '</a>';
695 },
696 // Цвет текста
697 '#\[color=(.*?)\](.*?)\[/color\]#si' => fn ($matches) => '<span style="color:' . self::clean_field($matches[1]) . ';">' . $process_recursively($matches[2], $depth + 1) . '</span>',
698 // Размер текста
699 '#\[size=(.*?)\](.*?)\[/size\]#si' => function ($matches) {
700 $size = (int)$matches[1];
701 $size = max(8, min(48, $size)); // Ограничение размера от 8px до 48px
702 return '<span style="font-size:' . $size . 'px;">' . $process_recursively($matches[2], $depth + 1) . '</span>';
703 },
704 // Цитирование
705 '#\[quote\](.*?)\[/quote\]#si' => fn ($matches) => '<blockquote>' . $process_recursively($matches[1], $depth + 1) . '</blockquote>',
706 '#\[quote=(.*?)\](.*?)\[/quote\]#si' => function ($matches) {
707 $author = self::clean_field($matches[1]); // Защита от XSS
708 return '<blockquote><strong>' . $author . ' писал:</strong><br />' . $process_recursively($matches[2], $depth + 1) . '</blockquote>';
709 },
710 // Списки
711 '#\[list\](.*?)\[/list\]#si' => fn ($matches) => '<ul>' . preg_replace('#\[\*\](.*?)#si', '<li>$1</li>', $process_recursively($matches[1], $depth + 1)) . '</ul>',
712 '#\[list=1\](.*?)\[/list\]#si' => fn ($matches) => '<ol type="1">' . preg_replace('#\[\*\](.*?)#si', '<li>$1</li>', $process_recursively($matches[1], $depth + 1)) . '</ol>',
713 '#\[list=a\](.*?)\[/list\]#si' => fn ($matches) => '<ol type="a">' . preg_replace('#\[\*\](.*?)#si', '<li>$1</li>', $process_recursively($matches[1], $depth + 1)) . '</ol>',
714 // Код
715 '#\[code\](.*?)\[/code\]#si' => fn ($matches) => '<pre><code>' . self::clean_field($matches[1]) . '</code></pre>',
716 // Спойлер
717 '#\[spoiler\](.*?)\[/spoiler\]#si' => fn ($matches) => '<details><summary>Показать/скрыть</summary>' . $process_recursively($matches[1], $depth + 1) . '</details>',
718 // Горизонтальная линия
719 '[hr]' => '<hr />',
720 // Перенос строки
721 '[br]' => '<br />',
722 // Выравнивание текста
723 '#\[left\](.*?)\[/left\]#si' => fn ($matches) => '<p style="text-align:left;">' . $process_recursively($matches[1], $depth + 1) . '</p>',
724 '#\[center\](.*?)\[/center\]#si' => fn ($matches) => '<p style="text-align:center;">' . $process_recursively($matches[1], $depth + 1) . '</p>',
725 '#\[right\](.*?)\[/right\]#si' => fn ($matches) => '<p style="text-align:right;">' . $process_recursively($matches[1], $depth + 1) . '</p>',
726 // Изображение
727 '#\[img\](.*?)\[/img\]#si' => function ($matches) {
728 $src = self::clean_field($matches[1]);
729 if (!filter_var($src, FILTER_VALIDATE_URL)) {
730 log_in_file(
731 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный URL изображения | Получено: '$src'"
732 );
733 return ''; // Удаляем некорректные изображения
734 }
735 return '<img src="' . $src . '" alt="' . $src . '" />';
736 },
737 ];
738 // Применяем паттерны к тексту
739 return preg_replace_callback_array($patterns, $text);
740 };
741 // Начинаем обработку с глубины 0
742 return $process_recursively($text, 0);
743 }

762 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
763 {
764 // Проверка граничных условий
765 if ($width <= 0 || empty($break)) {
766 log_in_file(
767 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = {$width}, break = '{$break}'"
768 );
769 return $str;
770 }
771 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
772 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
773 return $str;
774 }
775 // Разбиваем строку с использованием wordwrap()
776 $wrapped = wordwrap($str, $width, $break, true);
777 // Если строка содержит многобайтовые символы, корректируем разбиение
778 $lines = explode($break, $wrapped);
779 $result = [];
780 foreach ($lines as $line) {
781 if (mb_strlen($line, 'UTF-8') > $width) {
782 $result[] = mb_substr($line, 0, $width, 'UTF-8');
783 $result[] = mb_substr($line, $width, null, 'UTF-8');
784 } else {
785 $result[] = $line;
786 }
787 }
788 return implode($break, $result);
789 }
