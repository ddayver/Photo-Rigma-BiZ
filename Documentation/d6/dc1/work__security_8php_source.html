375 protected function _url_check_internal(): bool
376 {
377 if (!filter_var($_SERVER['REQUEST_URI'], FILTER_VALIDATE_URL)) {

465 protected function _check_input_internal(string $source_name, string $field, array $options = []): bool
466 {
467 $allowed_sources = ['_GET', '_POST', '_SESSION', '_COOKIE', '_FILES'];
468 if (!in_array($source_name, $allowed_sources, true)) {
469 log_in_file(
470 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый источник данных для check_input | Получено: {$source_name}"
471 );
472 return false;
473 }
474 $source = match ($source_name) {
475 '_GET' => $_GET,
476 '_POST' => $_POST,
477 '_SESSION' => $_SESSION,
478 '_COOKIE' => $_COOKIE,
479 '_FILES' => $_FILES,
480 default => null,
481 };
482 if (!is_array($source)) {
483 log_in_file(
484 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Источник данных не является массивом | Источник: {$source_name}"
485 );
486 return false;
487 }
488 if (($options['isset'] ?? false) && !isset($source[$field])) {
489 return false;
490 }
491 if (($options['empty'] ?? false) && empty($source[$field])) {
492 return false;
493 }
494 if ($source_name === '_FILES') {
495 if (!isset($source[$field])) {
496 return false;
497 }
498 $file = $source[$field];
499 $file_name = basename($file['name']);
500 if ($file['error'] !== UPLOAD_ERR_OK) {
501 log_in_file(
502 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка загрузки файла для поля | Поле: {$field}"
503 );
504 return false;
505 }
506 $max_size = $options['max_size'] ?? 0;
507 if ($max_size > 0 && $file['size'] > $max_size) {
508 log_in_file(
509 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Размер файла превышает максимально допустимый размер | Поле: {$field}, Размер: {$file['size']}, Максимальный размер: {$max_size}"
510 );
511 return false;
512 }
513 $finfo = finfo_open(FILEINFO_MIME_TYPE);
514 $real_mime_type = finfo_file($finfo, $file['tmp_name']);
515 finfo_close($finfo);
516 $is_mime_supported = Work::validate_mime_type($real_mime_type);
517 if (!$is_mime_supported) {
518 throw new \RuntimeException(
519 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Недопустимый MIME-тип для загруженного файла | Поле: {$field}, MIME-тип: {$real_mime_type}"
520 );
521 }
522 }
523 return $this->_check_field_internal(
524 $source[$field] ?? null,
525 $options['regexp'] ?? false,
526 $options['not_zero'] ?? false
527 );
528 }

546 protected function _check_field_internal(string $field, string|false $regexp = false, bool $not_zero = false): bool
547 {
548 if ($regexp !== false) {
549 $isValidRegexp = preg_match($regexp, '') !== false;
550 if (!$isValidRegexp) {
551 $errorMessage = match (preg_last_error()) {
552 PREG_INTERNAL_ERROR => "Внутренняя ошибка регулярного выражения.",
553 PREG_BACKTRACK_LIMIT_ERROR => "Превышен лимит обратного отслеживания.",
554 PREG_RECURSION_LIMIT_ERROR => "Превышен лимит рекурсии.",
555 PREG_BAD_UTF8_ERROR => "Некорректная UTF-8 строка.",
556 PREG_BAD_UTF8_OFFSET_ERROR => "Некорректный смещение UTF-8.",
557 default => "Неизвестная ошибка регулярного выражения.",
558 };
559 log_in_file(
560 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка в регулярном выражении | Регулярное выражение: {$regexp}, Причина: {$errorMessage}"
561 );
562 return false;
563 }
564 if (!preg_match($regexp, $field)) {
565 log_in_file(
566 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Значение поля не соответствует регулярному выражению | Поле: '{$field}', Регулярное выражение: {$regexp}"
567 );
568 return false;
569 }
570 }
571 if ($not_zero && $field === '0') {
572 log_in_file(
573 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Значение поля равно 0, но флаг not_zero установлен | Поле: '{$field}'"
574 );
575 return false;
576 }
577 foreach ($this->compiled_rules as $rule) {
578 if (preg_match($rule, $field)) {
579 log_in_file(
580 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Значение поля содержит запрещённый паттерн | Поле: '{$field}', Паттерн: {$rule}"
581 );
582 return false;
583 }
584 }
585 return true;
586 }

601 protected function _gen_captcha_internal(): array
602 {
603 $num1 = rand(1, 9);
604 $num2 = rand(1, 9);
605 $num3 = rand(1, 9);
606
607 $operation1 = rand(1, 2); // 1 - умножение, 2 - сложение
608 $operation2 = rand(1, 2); // 1 - умножение, 2 - сложение
609
610 $captcha = [
611 'question' => '',
612 'answer' => 0,
613 ];
614
615 if ($operation2 === 1) {
616 $captcha['question'] = "($num2 x $num3)";
617 $captcha['answer'] = $num2 * $num3;
618 } else {
619 $captcha['question'] = "($num2 + $num3)";
620 $captcha['answer'] = $num2 + $num3;
621 }
622
623 if ($operation1 === 1) {
624 $captcha['question'] = "$num1 x {$captcha['question']}";
625 $captcha['answer'] = $num1 * $captcha['answer'];
626 } else {
627 $captcha['question'] = "$num1 + {$captcha['question']}";
628 $captcha['answer'] = $num1 + $captcha['answer'];
629 }
630
631 return $captcha;
632 }
