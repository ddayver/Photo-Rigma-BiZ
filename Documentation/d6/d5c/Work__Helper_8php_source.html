509 protected static function _return_bytes_internal(string|int $val): int
510 {
511 // Проверяем, что входные данные являются строкой или числом
512 if (!is_string($val) && !is_numeric($val)) {
513 return 0; // Возвращаем 0 для некорректных данных
514 }
515 // Удаляем лишние пробелы
516 $val = trim((string)$val);
517 // Если строка содержит только цифры, возвращаем значение как есть
518 if (ctype_digit($val)) {
519 return (int)$val;
520 }
521 // Извлекаем последний символ (суффикс)
522 $last = strtolower($val[-1]);
523 // Извлекаем числовую часть (все символы, кроме последнего)
524 $number = substr($val, 0, -1);
525 // Проверяем, является ли числовая часть корректной
526 if (!is_numeric($number)) {
527 return 0; // Возвращаем 0 для некорректных данных
528 }
529 // Преобразуем в абсолютное значение
530 $number = abs((float)$number);
531 // Преобразуем в байты с использованием match
532 return match ($last) {
533 'g' => (int)($number * 1024 * 1024 * 1024),
534 'm' => (int)($number * 1024 * 1024),
535 'k' => (int)($number * 1024),
536 default => (int)$number, // Если суффикс недопустим, возвращаем только числовую часть
537 };

994 */
995 protected static function _utf8_wordwrap_internal(string $str, int $width = 70, string $break = PHP_EOL): string
996 {
997 // Проверка граничных условий
998 if ($width <= 0 || empty($break)) {
999 log_in_file(
1000 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректные параметры | width = $width, break = '$break'"
1001 );
1002 return $str;
1003 }
1004 // Если строка пустая или её длина меньше или равна $width, возвращаем её без изменений
1005 if (empty($str) || mb_strlen($str, 'UTF-8') <= $width) {
1006 return $str;
1007 }
1008 // Разбиваем строку с использованием wordwrap()
1009 $wrapped = wordwrap($str, $width, $break, true);
1010 // Если строка содержит многобайтовые символы, корректируем разбиение
1011 $lines = explode($break, $wrapped);
1012 $result = [];
1013 foreach ($lines as $line) {
1014 if (mb_strlen($line, 'UTF-8') > $width) {
1015 $result[] = mb_substr($line, 0, $width, 'UTF-8');
1016 $result[] = mb_substr($line, $width, null, 'UTF-8');
1017 } else {
1018 $result[] = $line;
1019 }
1020 }
1021 return implode($break, $result);
1022 }
