228 private function test_is_object(string $path_array): array
229 {
230 // Разбиваем путь на части по разделителю '->'
231 $tmp_path = explode('->', $path_array);
232 $tmp_p = $tmp_path[0]; // Берем первую часть пути
233 $tmp_p = strtoupper($tmp_p); // Приводим к верхнему регистру
234
235 // Разбиваем имя массива и индекс по символу '['
236 $tmp = explode('[', $tmp_p);
237
238 // Проверяем корректность имени массива и индекса
239 if (mb_ereg('^[A-Z_]+$', $tmp[0]) && isset($tmp[1]) && !empty($tmp[1])) {
240 $tmp[1] = str_replace(']', '', $tmp[1]); // Убираем закрывающую скобку ']'
241 if (!mb_ereg('^[0-9]+$', $tmp[1])) {
242 log_in_file('Error in Path OBJ::' . $path_array, DIE_IF_ERROR);
243 }
244 } else {
245 log_in_file('Error in Path OBJ::' . $path_array, DIE_IF_ERROR);
246 }
247
248 // Проверяем, существует ли объект в массиве block_object
249 if (!isset($this->block_object[$tmp[0]][$tmp[1]]) || !is_object($this->block_object[$tmp[0]][$tmp[1]])) {
250 // Если объект не существует, создаем новый экземпляр класса template
251 $temp = new template($this->site_url, $this->site_dir, $this->theme);
252 $this->block_object[$tmp[0]][$tmp[1]] = $temp;
253 }
254
255 // Формируем результат
256 $result['current'] = $tmp[0]; // Имя массива
257 $result['index'] = $tmp[1]; // Индекс элемента
258 if (count($tmp_path) > 1) {
259 // Если есть продолжение пути, сохраняем его
260 unset($tmp_path[0]);
261 $result['next_path'] = implode('->', $tmp_path);
262 } else {
263 // Если путь полностью разобран, возвращаем FALSE
264 $result['next_path'] = false;
265 }
266
267 return $result;
268 }

332 private function template_if($key, $val)
333 {
334 $pattern = '/<!-- IF_' . preg_quote($key, '/') . '_BEGIN -->(.*?)<!-- IF_' . preg_quote($key, '/') . '_ELSE -->(.*?)<!-- IF_' . preg_quote($key, '/') . '_END -->/s';
335 if (preg_match_all($pattern, $this->content, $matches)) {
336 foreach ($matches[0] as $i => $match) {
337 $temp_content = $matches[0][$i];
338 $if_block = $matches[1][$i];
339 $else_block = $matches[2][$i];
340
341 // Выбираем блок в зависимости от значения $val
342 $replacement = $val ? $if_block : $else_block;
343
344 // Заменяем исходный блок на выбранный
345 $this->content = str_replace($temp_content, $replacement, $this->content);
346 }
347 } else {
348 // Если нет ELSE, обрабатываем только BEGIN-END
349 $pattern_no_else = '/<!-- IF_' . preg_quote($key, '/') . '_BEGIN -->(.*?)<!-- IF_' . preg_quote($key, '/') . '_END -->/s';
350 if (preg_match_all($pattern_no_else, $this->content, $matches_no_else)) {
351 foreach ($matches_no_else[0] as $i => $match) {
352 $temp_content = $matches_no_else[0][$i];
353 $if_block = $matches_no_else[1][$i];
354
355 // Выбираем блок в зависимости от значения $val
356 $replacement = $val ? $if_block : '';
357
358 // Заменяем исходный блок на выбранный

373 private function template_case($key, $val)
374 {
375 $pattern = '/<!-- SELECT_' . preg_quote($key, '/') . '_BEGIN -->(.*?)<!-- CASE_(DEFAULT|[A-Z0-9_]+) -->(.*?)<!-- BREAK -->(.*?)<!-- SELECT_' . preg_quote($key, '/') . '_END -->/s';
376 if (preg_match_all($pattern, $this->content, $matches)) {
377 foreach ($matches[0] as $i => $match) {
378 $temp_content = $matches[0][$i];
379 $case_value = $matches[2][$i];
380 $case_block = $matches[3][$i];
381 $default_block = null;
382
383 // Проверяем, есть ли блок DEFAULT
384 if ($case_value === 'DEFAULT') {
385 $default_block = $case_block;
386 continue; // Пропускаем DEFAULT при первом проходе
387 }
388
389 // Если значение совпадает, выбираем соответствующий блок
390 if (strtoupper($val) === $case_value) {
391 $this->content = str_replace($temp_content, $case_block, $this->content);
392 return;
393 }
394 }
395
396 // Если ни один блок не совпал, используем DEFAULT
397 if ($default_block !== null) {
398 $this->content = str_replace($temp_content, $default_block, $this->content);
399 } else {
400 log_in_file('Error template SELECT-CASE: No matching case or default block for ' . $key, DIE_IF_ERROR);
401 }
402 } else {
403 log_in_file('Error template SELECT-CASE: ' . $key, DIE_IF_ERROR);
404 }
