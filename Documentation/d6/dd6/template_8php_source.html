570 private function test_is_object(string $path_array): array
571 {
572 // Разбор пути
573 $path_parts = explode('->', $path_array);
574 $first_part = strtoupper($path_parts[0]);
575 $first_part_details = explode('[', $first_part);
576 // Проверка корректности первого элемента
577 if (empty($first_part_details[1]) || !preg_match(
578 '/^[A-Z_]+$/',
579 $first_part_details[0]
580 )) {
581 throw new InvalidArgumentException(
582 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка в пути объекта | Путь: $path_array"
583 );
584 }
585 // Проверка индекса
586 $index = str_replace(']', '', $first_part_details[1]);
587 if (!preg_match('/^\d+$/', $index)) {
588 throw new InvalidArgumentException(
589 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный индекс в пути | Путь: $path_array"
590 );
591 }
592 // Создание объекта, если он не существует
593 if (!isset($this->block_object[$first_part_details[0]][$index]) || !is_object(
594 $this->block_object[$first_part_details[0]][$index]
595 )) {
596 $this->block_object[$first_part_details[0]][$index] = new Template(
597 $this->site_url,
598 $this->site_dir,
599 $this->theme
600 );
601 }
602 // Формирование результата
603 return [
604 'current' => $first_part_details[0],
605 'index' => $index,
606 'next_path' => count($path_parts) > 1 ? implode('->', array_slice($path_parts, 1)) : false,
607 ];
608 }
