870 private function test_is_object(string $path_array): array
871 {
872 // Разбор пути
873 $path_parts = explode('->', $path_array);
874 $first_part = strtoupper($path_parts[0]);
875 $first_part_details = explode('[', $first_part);
876 // Проверка корректности первого элемента
877 if (empty($first_part_details[1]) || !preg_match(
878 '/^[A-Z_]+$/',
879 $first_part_details[0]
880 )) {
881 throw new InvalidArgumentException(
882 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка в пути объекта | Путь: $path_array"
883 );
884 }
885 // Проверка индекса
886 $index = str_replace(']', '', $first_part_details[1]);
887 if (!preg_match('/^\d+$/', $index)) {
888 throw new InvalidArgumentException(
889 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный индекс в пути | Путь: $path_array"
890 );
891 }
892 // Создание объекта, если он не существует
893 if (!isset($this->block_object[$first_part_details[0]][$index]) || !is_object(
894 $this->block_object[$first_part_details[0]][$index]
895 )) {
896 $this->block_object[$first_part_details[0]][$index] = new Template(
897 $this->site_url,
898 $this->site_dir,
899 $this->theme
900 );
901 }
902 // Формирование результата
903 return [
904 'current' => $first_part_details[0],
905 'index' => $index,
906 'next_path' => count($path_parts) > 1 ? implode('->', array_slice($path_parts, 1)) : false,
907 ];
908 }
