874 private function _test_is_object(string $path_array): array
875 {
876 // Разбор пути
877 $path_parts = explode('->', $path_array);
878 $first_part = strtoupper($path_parts[0]);
879 $first_part_details = explode('[', $first_part);
880 // Проверка корректности первого элемента
881 if (empty($first_part_details[1]) || !preg_match(
882 '/^[A-Z_]+$/',
883 $first_part_details[0]
884 )) {
885 throw new InvalidArgumentException(
886 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Ошибка в пути объекта | Путь: $path_array"
887 );
888 }
889 // Проверка индекса
890 $index = str_replace(']', '', $first_part_details[1]);
891 if (!preg_match('/^\d+$/', $index)) {
892 throw new InvalidArgumentException(
893 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный индекс в пути | Путь: $path_array"
894 );
895 }
896 // Создание объекта, если он не существует
897 if (!isset($this->block_object[$first_part_details[0]][$index]) || !is_object(
898 $this->block_object[$first_part_details[0]][$index]
899 )) {
900 $this->block_object[$first_part_details[0]][$index] = new Template(
901 $this->site_url,
902 $this->site_dir,
903 $this->theme
904 );
905 }
906 // Формирование результата
907 return [
908 'current' => $first_part_details[0],
909 'index' => $index,
910 'next_path' => count($path_parts) > 1 ? implode('->', array_slice($path_parts, 1)) : false,
911 ];
912 }
