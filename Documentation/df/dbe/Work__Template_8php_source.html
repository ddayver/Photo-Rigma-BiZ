435 protected function _create_menu_internal(string $action, int $menu): array
436 {
437 // Валидация входных данных.
438 if (!in_array($menu, [SHORT_MENU, LONG_MENU], true)) {
439 throw new InvalidArgumentException(
440 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
441 );
442 }
443 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
444 $menu_type = match ($menu) {
445 SHORT_MENU => 'short',
446 LONG_MENU => 'long',
447 default => throw new InvalidArgumentException(
448 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
449 ),
450 };
451 // Формирование запроса для получения данных меню (используем параметры для безопасности)
452 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
453 $this->db->select(
454 '*', // Все поля таблицы используются для этого запроса
455 TBL_MENU,
456 [
457 'where' => '`' . $menu_type . '` = :menu_type',
458 'params' => [':menu_type' => 1],
459 'order' => '`id` ASC',
460 ]
461 );
462 // Получение результатов запроса
463 $menu_data = $this->db->result_array();
464 if ($menu_data === false) {
465 throw new RuntimeException(
466 __FILE__ . ':' . __LINE__ . ' (' . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ') | Не удалось получить данные из базы данных'
467 );
468 }
469
470 $menu_items = [];
471 foreach ($menu_data as $key => $menu_item) {
472 $is_visible = true;
473
474 // Проверяем права доступа для текущего пользователя
475 if ($menu_item['user_login'] !== null) {
476 // user_login: 0 - только гостям, 1 - только зарегистрированным
477 if (($menu_item['user_login'] === 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
478 ($menu_item['user_login'] === 1 && $this->user->user['id'] === 0) // Зарегистрированный пытается получить доступ к пункту для гостей
479 ) {
480 $is_visible = false;
481 }
482 }
483
484 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
485 if (($menu_item['user_access'] !== null) && empty($this->user->user[$menu_item['user_access']])) {
486 $is_visible = false;
487 }
488 // Если пункт меню видим, добавляем его в результат
489 if ($is_visible) {
490 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
491 $url_action = Work::clean_field($menu_item['url_action']);
492 $name_action = Work::clean_field($menu_item['name_action']);
493 $menu_items[$key] = [
494 'url' => ($menu_item['action'] === $action ? null : $this->config['site_url'] . $url_action),
495 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action),
496 ];
497 }
498 }
499 return $menu_items;
500 }
