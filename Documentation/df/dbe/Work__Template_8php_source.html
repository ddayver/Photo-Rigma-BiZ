436 protected function _create_menu_internal(string $action, int $menu): array
437 {
438 // Валидация входных данных.
439 if (!in_array($menu, [SHORT_MENU, LONG_MENU], true)) {
440 throw new InvalidArgumentException(
441 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
442 );
443 }
444 // Определение типа меню через match() (поддерживается начиная с PHP 8.0)
445 $menu_type = match ($menu) {
446 SHORT_MENU => 'short',
447 LONG_MENU => 'long',
448 default => throw new InvalidArgumentException(
449 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Некорректный тип меню (\$menu)"
450 ),
451 };
452 // Формирование запроса для получения данных меню (используем параметры для безопасности)
453 // Условие ":menu_type = 1" используется для выборки по имени столбца (смена имени столбца для выборки).
454 $this->db->select(
455 '*', // Все поля таблицы используются для этого запроса
456 TBL_MENU,
457 [
458 'where' => '`' . $menu_type . '` = :menu_type',
459 'params' => [':menu_type' => 1],
460 'order' => '`id` ASC',
461 ]
462 );
463 // Получение результатов запроса
464 $menu_data = $this->db->res_arr();
465 if ($menu_data === false) {
466 throw new RuntimeException(
467 __FILE__ . ":" . __LINE__ . " (" . (__METHOD__ ?: __FUNCTION__ ?: 'global') . ") | Не удалось получить данные из базы данных"
468 );
469 }
470
471 $menu_items = [];
472 foreach ($menu_data as $key => $menu_item) {
473 $is_visible = true;
474
475 // Проверяем права доступа для текущего пользователя
476 if ($menu_item['user_login'] !== null) {
477 // user_login: 0 - только гостям, 1 - только зарегистрированным
478 if (($menu_item['user_login'] === 0 && $this->user->user['id'] > 0) || // Гость пытается получить доступ к пункту для зарегистрированных
479 ($menu_item['user_login'] === 1 && $this->user->user['id'] === 0) // Зарегистрированный пытается получить доступ к пункту для гостей
480 ) {
481 $is_visible = false;
482 }
483 }
484
485 // user_access: NULL - всем, иначе проверяем значение в $this->user->user
486 if (($menu_item['user_access'] !== null) && empty($this->user->user[$menu_item['user_access']])) {
487 $is_visible = false;
488 }
489 // Если пункт меню видим, добавляем его в результат
490 if ($is_visible) {
491 // Очищаем данные с использованием Work::clean_field() (статический метод с расширенными проверками)
492 $url_action = Work::clean_field($menu_item['url_action']);
493 $name_action = Work::clean_field($menu_item['name_action']);
494 $menu_items[$key] = [
495 'url' => ($menu_item['action'] === $action ? null : $this->config['site_url'] . $url_action),
496 'name' => $this->lang['menu'][$name_action] ?? ucfirst($name_action),
497 ];
498 }
499 }
500 return $menu_items;
501 }

707 protected function _template_best_user_internal(int $best_user = 1): array
708 {
709 // Блок получения данных через join()
710 $this->db->join(
711 [TBL_USERS . '.`id`', TBL_USERS . '.`real_name`', 'COUNT(' . TBL_PHOTO . '.`id`) AS `user_photo`'],
712 TBL_USERS, // Основная таблица (user)
713 [
714 [
715 'type' => 'INNER', // Тип JOIN
716 'table' => TBL_PHOTO, // Присоединяемая таблица (photo)
717 'on' => TBL_USERS . '.`id` = ' . TBL_PHOTO . '.`user_upload`', // Условие JOIN
718 ],
719 ],
720 [
721 'group' => TBL_USERS . '.`id`', // Группировка по ID пользователя
722 'order' => '`user_photo` DESC', // Сортировка по количеству фотографий
723 'limit' => $best_user, // Лимит на количество результатов
724 ]
725 );
726 $user_data = $this->db->res_arr();
727 // Проверка: $user_data может быть массивом (если данные есть) или false (если запрос не вернул данных).
728 // Обработка данных пользователей
729 $array_data = ($user_data !== false) ? array_map(function ($current_user) {
730 return [
731 'user_url' => sprintf(
732 '%s?action=profile&amp;subact=profile&amp;uid=%d',
733 $this->config['site_url'],
734 $current_user['id']
735 ),
736 'user_name' => Work::clean_field($current_user['real_name']),
737 'user_photo' => (int)$current_user['user_photo'],
738 ];
739 }, $user_data) : [];
740
741 // Добавление данных для пустого пользователя
742 if (empty($array_data)) {
743 $array_data[1] = [
744 'user_url' => '',
745 'user_name' => '---',
746 'user_photo' => '-',
747 ];
748 }
749
750 // Блок формирования метаданных для шаблона
751 $top_data = [
752 'NAME_BLOCK' => sprintf($this->lang['main']['best_user'], $best_user),
753 'L_USER_NAME' => $this->lang['main']['user_name'],
754 'L_USER_PHOTO' => $this->lang['main']['best_user_photo'],
755 ];
756 array_unshift($array_data, $top_data);
757
758 return $array_data;
