185  function utf8_wordwrap($str, $width = 70, $break = PHP_EOL)
186  {
187  if (empty($str) || mb_strlen($str, 'UTF-8') <= $width)
188  return $str;
189  $br_width = mb_strlen($break, 'UTF-8');
190  $str_width = mb_strlen($str, 'UTF-8');
191  $return = '';
192  $last_space = false;
193  for ($i = 0, $count = 0; $i < $str_width; $i++, $count++)
194  {
195  if (mb_substr($str, $i, $br_width, 'UTF-8') == $break)
196  {
197  $count = 0;
198  $return .= mb_substr($str, $i, $br_width, 'UTF-8');
199  $i += $br_width - 1;
200  continue;
201  }
202  if (mb_substr($str, $i, 1, 'UTF-8') == " ")
203  {
204  $last_space = $i;
205  }
206  if ($count > $width)
207  {
208  if (!$last_space)
209  {
210  $return .= $break;
211  $count = 0;
212  }
213  else
214  {
215  $drop = $i - $last_space;
216  if ($drop > 0)
217  {
218  $return = mb_substr($return, 0, -$drop);
219  }
220  $return .= $break;
221  $i = $last_space + ($br_width - 1);
222  $last_space = false;
223  $count = 0;
224  }
225  }
226  $return .= mb_substr($str, $i, 1, 'UTF-8');
227  }
228  return $return;
