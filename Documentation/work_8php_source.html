185  function utf8_wordwrap($str, $width = 70, $break = PHP_EOL)
186  {
187  if (empty($str) || mb_strlen($str, 'UTF-8') <= $width)
188  return $str;
189  $br_width = mb_strlen($break, 'UTF-8');
190  $str_width = mb_strlen($str, 'UTF-8');
191  $return = '';
192  $last_space = false;
193  for ($i = 0, $count = 0; $i < $str_width; $i++, $count++)
194  {
195  if (mb_substr($str, $i, $br_width, 'UTF-8') == $break)
196  {
197  $count = 0;
198  $return .= mb_substr($str, $i, $br_width, 'UTF-8');
199  $i += $br_width - 1;
200  continue;
201  }
202  if (mb_substr($str, $i, 1, 'UTF-8') == " ")
203  {
204  $last_space = $i;
205  }
206  if ($count > $width)
207  {
208  if (!$last_space)
209  {
210  $return .= $break;
211  $count = 0;
212  }
213  else
214  {
215  $drop = $i - $last_space;
216  if ($drop > 0)
217  {
218  $return = mb_substr($return, 0, -$drop);
219  }
220  $return .= $break;
221  $i = $last_space + ($br_width - 1);
222  $last_space = false;
223  $count = 0;
224  }
225  }
226  $return .= mb_substr($str, $i, 1, 'UTF-8');
227  }
228  return $return;
229  }

766  function size_image($path_image)
767  {
768  $size = getimagesize($path_image);
769  if ($this->config['temp_photo_w'] == '0') $ratio_width = 1;
770  else $ratio_width = $size[0]/$this->config['temp_photo_w'];
771  if ($this->config['temp_photo_h'] == '0') $ratio_height = 1;
772  else $ratio_height = $size[1]/$this->config['temp_photo_h'];
773 
774  if($size[0] < $this->config['temp_photo_w'] && $size[1] < $this->config['temp_photo_h'] && $this->config['temp_photo_w'] != '0' && $this->config['temp_photo_h'] != '0')
775  {
776  $size_photo['width'] = $size[0];
777  $size_photo['height'] = $size[1];
778  }
779  else
780  {
781  if($ratio_width < $ratio_height)
782  {
783  $size_photo['width'] = $size[0]/$ratio_height;
784  $size_photo['height'] = $size[1]/$ratio_height;
785  }
786  else
787  {
788  $size_photo['width'] = $size[0]/$ratio_width;
789  $size_photo['height'] = $size[1]/$ratio_width;
790  }
791  }
792  return $size_photo;
